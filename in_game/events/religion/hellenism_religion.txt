hellenism_religion.1 = { # <Selected God> Smiles Upon Us
	type = country_event
	title = hellenism_religion.1.title
	desc = hellenism_religion.1.desc
	outcome = neutral

	illustration_tags = {
		10 = happy
		10 = exterior
	}

	trigger = {
		current_date > 476.9.4
		has_dlc = "d008_fate_of_the_phoenix"
		religion = religion:hellenism_religion
		num_omens > 0
		at_war = no
		NOT = { has_religious_aspect = religious_aspect:monism_aspect }
	}

	immediate = {
		event_illustration_religion_effect = { religion = religion:hellenism_religion } 

		random_current_gods = { save_scope_as = target_god }

		ruler_or_regent = { save_scope_as = target_character }

		# Pick a random "blessing variant" based on the selected god
		switch = {
			trigger = scope:target_god

			# ---------------------------
			# ZEUS
			# ---------------------------
			god:zeus_god = {
				random_list = {
					1 = { set_variable = hellenism_religion_zeus_1 }
					1 = { set_variable = hellenism_religion_zeus_2 }
				}
			}

			# ---------------------------
			# APHRODITE
			# ---------------------------
			god:aphrodite_god = {
				random_list = {
					1 = { set_variable = hellenism_religion_aphrodite_1 }
					1 = { set_variable = hellenism_religion_aphrodite_2 }
				}
			}

			# ---------------------------
			# POSEIDON
			# ---------------------------
			god:poseidon_god = {
				random_list = {
					1 = { set_variable = hellenism_religion_poseidon_1 }
					1 = { set_variable = hellenism_religion_poseidon_2 }
				}
			}

			# ---------------------------
			# HERMES
			# ---------------------------
			god:hermes_god = {
				random_list = {
					1 = { set_variable = hellenism_religion_hermes_1 }
					1 = { set_variable = hellenism_religion_hermes_2 }
				}
			}

			# ---------------------------
			# ARES
			# ---------------------------
			god:ares_god = {
				random_list = {
					1 = { set_variable = hellenism_religion_ares_1 }
					1 = { set_variable = hellenism_religion_ares_2 }
				}
			}

			# ---------------------------
			# ATHENA
			# ---------------------------
			god:athena_god = {
				random_list = {
					1 = { set_variable = hellenism_religion_athena_1 }
					1 = { set_variable = hellenism_religion_athena_2 }
				}
			}
		}
	}

	option = {
		name = hellenism_religion.1.a

		# Apply the chosen blessing variant
		switch = {
			trigger = has_variable

			# ==================================================
			# ZEUS
			# ==================================================
			hellenism_religion_zeus_1 = {
				# Ruler ADM
				scope:target_character = { add_adm = { 1 5 } }
			}
			hellenism_religion_zeus_2 = {
				# Government Power
				add_government_power = government_power_mild_bonus
			}

			# ==================================================
			# APHRODITE
			# ==================================================
			hellenism_religion_aphrodite_1 = {
				# Stability
				add_stability = stability_mild_bonus
			}
			hellenism_religion_aphrodite_2 = {
				# Fertility
				custom_tooltip = {
					text = fertile_ruler_and_family_tt
					scope:target_character = { add_fertility = 20 }
					consort ?= { add_fertility = 20 }
				}
			}

			# ==================================================
			# POSEIDON
			# ==================================================
			hellenism_religion_poseidon_1 = {
				# Improve a few key ports (development)
				ordered_port_in_country = {
					order_by = {
						value = population
						multiply = local_control
						multiply = local_control
					}
					max = {
						value = 5
						multiply = root.religious_unity
					}
					check_range_bounds = no
					change_development = development_weak_bonus
				}
			}
			hellenism_religion_poseidon_2 = {
				# Improve a few key ports (prosperity)
				ordered_port_in_country = {
					order_by = {
						value = population
						multiply = local_control
						multiply = local_control
					}
					max = {
						value = 5
						multiply = root.religious_unity
					}
					check_range_bounds = no
					change_prosperity = prosperity_weak_bonus
				}
			}

			# ==================================================
			# HERMES
			# ==================================================
			hellenism_religion_hermes_1 = {
				# Burghers Satisfaction
				add_estate_satisfaction = {
					type = estate_type:burghers_estate
					value = estate_satisfaction_extreme_bonus
				}
			}
			hellenism_religion_hermes_2 = {
				# Dignitaries / coalition softening (fallback: ruler ADM)
				if = {
					limit = {
						any_country_with_coalition_grade_antagonism_against_us = { count > 4 }
					}
					ordered_country_with_coalition_grade_antagonism_against_us = {
						order_by = total_population
						max = 4
						check_range_bounds = no
						add_opinion = { target = root modifier = opinion_dignitaries_of_hermes }
						add_antagonism = { target = root modifier = dignitaries_of_hermes_antagonism }
					}
				}
				else = {
					scope:target_character = { add_adm = { 1 5 } }
				}
			}

			# ==================================================
			# ARES
			# ==================================================
			hellenism_religion_ares_1 = {
				# Ruler MIL
				scope:target_character = { add_mil = { 1 5 } }
			}
			hellenism_religion_ares_2 = {
				# Army Tradition
				add_army_tradition = army_tradition_severe_bonus
			}

			# ==================================================
			# ATHENA
			# ==================================================
			hellenism_religion_athena_1 = {
				# Research Progress
				add_research_progress = research_progress_mild_bonus
			}
			hellenism_religion_athena_2 = {
				# Cultural Influence
				culture = { add_cultural_influence = cultural_influence_mild_bonus }
			}
		}
	}

	after = {
		hidden_effect = {

			# ZEUS
			remove_variable = hellenism_religion_zeus_1
			remove_variable = hellenism_religion_zeus_2

			# APHRODITE
			remove_variable = hellenism_religion_aphrodite_1
			remove_variable = hellenism_religion_aphrodite_2

			# POSEIDON
			remove_variable = hellenism_religion_poseidon_1
			remove_variable = hellenism_religion_poseidon_2

			# HERMES
			remove_variable = hellenism_religion_hermes_1
			remove_variable = hellenism_religion_hermes_2

			# ARES
			remove_variable = hellenism_religion_ares_1
			remove_variable = hellenism_religion_ares_2

			# ATHENA
			remove_variable = hellenism_religion_athena_1
			remove_variable = hellenism_religion_athena_2
		}
	}
}


hellenism_religion.2 = { # The Eagle of <General/Admiral>
	type = country_event
	title = hellenism_religion.2.title
	desc = hellenism_religion.2.desc
	outcome = neutral

	illustration_tags = {
		10 = military
		
		10 = exterior
	}

	trigger = {
		current_date > 476.9.4
		has_dlc = "d008_fate_of_the_phoenix"
		religion = religion:hellenism_religion
		OR = {
			any_army = { 
				has_commander = yes 
				leader = { 
					has_nickname = no
					total_abilities > 150
				}
			}
			any_navy = { 
				has_commander = yes 
				leader = {
					has_nickname = no
					total_abilities > 150
				}
			}
		}
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }

		ordered_unit = {
			limit = { 
				has_commander = yes
				leader = { 
					has_nickname = no
					total_abilities > 150
				}
			}
			order_by = {
				value = unit_strength
				if = {
					limit = { has_commander = yes }
					add = leader.total_abilities
				}
			}
			max = 3
			check_range_bounds = no
			add_to_list = hellenism_religion_2_list
		}

		random_in_list = {
			list = hellenism_religion_2_list

			save_scope_as = target_army
			leader = { save_scope_as = target_character }
		}
	}

	option = { 
		name = hellenism_religion.2.a 
	
		scope:target_character = {
			if = {
				limit = {
					is_general = yes
				}
				if = {
					limit = { NOT = { has_trait_category = general } }
					add_random_trait_from_category = general
				}
				else = {
					add_adm = { 1 5 }
					add_dip = { 1 5 }
					add_mil = { 1 5 }
				}
				root = { add_army_tradition = army_tradition_mild_bonus }
			}
			else = { 
				if = {
					limit = { NOT = { has_trait_category = admiral } }
					add_random_trait_from_category = admiral
				} 
				else = {
					add_adm = { 1 5 }
					add_dip = { 1 5 }
					add_mil = { 1 5 }
				}
				root = { add_navy_tradition = navy_tradition_mild_bonus }
			}

			if = {
				limit = { has_nickname = no }
				set_nickname = eagle_nickname
			}
		}
	}

	option = {
		name = hellenism_religion.2.b 

		if = {
			limit = {
				scope:target_character = { is_general = yes }
			}

			change_societal_value = {
				type = land_vs_naval
				value = societal_value_move_to_left
			}
		}
		else = {
			change_societal_value = {
				type = land_vs_naval
				value = societal_value_move_to_right
			}
		}

		add_research_progress = research_progress_ultimate_bonus

		remove_commander = scope:target_character
		scope:target_character = {
			add_character_modifier = {
				modifier = block_leading_armies_or_navies
				years = modifier_duration_years_normal
				mode = add_and_extend
			}
		}
	}
}
