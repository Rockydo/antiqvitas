imperial_circles.2 = { #Missing Funds - Imperial Bribe is not distributed in time
	type = country_event
	title = imperial_circles.2.title
	desc = imperial_circles.2.desc
	outcome = neutral
	category = international_organization_event
	image = "gfx/interface/illustrations/international_organization_types/hre.dds"

	immediate = {
		save_scope_as = actor
	}

	option = { #executes distribute_imperial_grant effect without granting the Circle Satisfaction
		name = imperial_circles.2.a

		custom_tooltip = {
				text = distribute_imperial_grant_tt

				scope:actor = {
					save_temporary_scope_value_as = {
						name = total_grant
						value = {
							value = country_economical_base
							multiply = 0.20
						}
					}
				}
				every_imperial_circle_country_is_member_of = {
					save_temporary_scope_value_as = {
						name = grant_per_member
						value = {
							value = scope:total_grant
							divide = {
								value = imperial_circle_member_count
								subtract = 1
							}
						}
					}

					every_circle_member = {
						limit = {
							NOT = { this = scope:actor }
						}
						add_gold = scope:grant_per_member
						scope:actor = {
							add_gold = {
								value = scope:grant_per_member
								multiply = -1
							}
						}
						add_country_modifier = {
							modifier = received_imperial_circle_grant
							years = 2
						}
						add_opinion = {
							target = scope:actor
							modifier = opinion_received_imperial_grant
						}
					}
				}
			}
	}

	option = { #hurst circle satisfaction
		name = imperial_circles.2.b

		every_imperial_circle_country_is_member_of = {
				add_circle_satisfaction = -5
			}
	}

	option = { #very good relations with the emperor prevent damage
		name = imperial_circles.2.c

		trigger = {
			international_organization:hre = {
				international_organization_has_leader = yes
			}
			is_emperor = no
			international_organization:hre.leader_country = {
				opinion = {
					target = root
					value >= 150
				}
			}
		}

		add_prestige = prestige_mild_bonus
	}

	after = {
		remove_variable = has_received_imperial_bribe
	}

	trigger = {
		current_date > 476.9.4
	}
}

imperial_circles.50 = { #Challenge for Circle Leadership
	type = country_event
	title = imperial_circles.50.title
	desc = imperial_circles.50.desc
	image = "gfx/interface/illustrations/international_organization_types/hre.dds"
	outcome = neutral

	option = {
		name = imperial_circles.50.a		# Press the challenge
		if = {
			limit = {
				OR = {
					any_imperial_circle_country_is_member_of = {
						imperial_circle_satisfaction < 15
					}
					AND = {
						any_imperial_circle_country_is_member_of = {
							imperial_circle_satisfaction < 30
						}
						prestige >= scope:recipient.prestige
					}
					prestige >= {
						value = scope:recipient.prestige
						add = 25
					}
				}
			}
			every_imperial_circle_country_is_member_of = {
				set_circle_leader = scope:actor
			}
			add_country_modifier = {
				modifier = consolidating_circle_leadership
				years = 2
			}
			scope:recipient = {
				add_prestige = prestige_mild_penalty
				trigger_event_silently = { id = imperial_circles.51 }
			}
		}
		else = {
			add_prestige = prestige_mild_penalty
			scope:recipient = {
				trigger_event_silently = { id = imperial_circles.52 }
			}
			trigger_event_silently = { id = imperial_circles.53 }
		}
	}

	option = {
		name = imperial_circles.50.b		# Withdraw the challenge
		add_prestige = prestige_weak_penalty
	}
	trigger = {
		current_date > 476.9.4
	}
}
