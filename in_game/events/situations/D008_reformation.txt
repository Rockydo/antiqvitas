namespace = reformation

# Melanchthon's Letter to Constantinople
# Triggered by the contact_patriarch_of_constantinople generic action
reformation.17 = {
	type = country_event
	category = situation_event
	title = reformation.17.title
	desc = reformation.17.desc
	outcome = neutral
	historical_info = reformation.17.historical_info

	illustration_tags = {
		10 = regular
		10 = interior
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:burghers_estate }
		save_scope_as = from
		character:reformer_philip_melanchton = {
			save_scope_as = melanchthon
		}
		location:constantinople.owner = {
			save_scope_as = patriarch_country
		}
	}

	option = {
		name = reformation.17.a
		add_prestige = prestige_weak_penalty
		set_variable = lutheran_sent_delegation
		scope:patriarch_country = {
			trigger_event_non_silently = { id = reformation.18 days = { 60 90 } }
		}
	}

	option = {
		name = reformation.17.b
		add_religious_influence = religious_influence_weak_penalty
		location:constantinople.owner = {
			if = {
				limit = { has_societal_value = societal_value_type:latinization_vs_hellenization }
				change_societal_value = { type = latinization_vs_hellenization value = societal_value_minor_move_to_right }
			}
		}
	}
	trigger = {
		current_date > 476.9.4
	}
}

# A Letter from the Lutheran Reformers
# Fires at the Orthodox country that owns Constantinople
reformation.18 = {
	type = country_event
	category = situation_event
	title = reformation.18.title
	desc = reformation.18.desc
	outcome = neutral
	historical_info = reformation.18.historical_info

	trigger = {
		current_date > 476.9.4
		exists = scope:from
		location:constantinople = {
			has_owner = yes
			owner = { this = root }
		}
	}

	illustration_tags = {
		10 = regular
		10 = interior
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
		save_scope_as = patriarch_country
		scope:from = {
			save_scope_as = lutheran_country
		}
	}

	option = {
		name = reformation.18.a
		scope:from = {
			trigger_event_non_silently = { id = reformation.19 days = 60 }
		}
	}

	option = {
		name = reformation.18.b
		scope:from = {
			trigger_event_non_silently = { id = reformation.20 days = 30 }
		}
	}
}

# The Patriarch Responds with Interest (positive branch)
reformation.19 = {
	type = country_event
	category = situation_event
	title = reformation.19.title
	desc = reformation.19.desc
	outcome = positive
	historical_info = reformation.19.historical_info

	trigger = {
		current_date > 476.9.4
		has_variable = lutheran_sent_delegation
	}

	illustration_tags = {
		10 = happy
		10 = interior
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
	}

	option = {
		name = reformation.19.a
		add_religious_influence = religious_influence_mild_bonus
		set_variable = lutheran_orthodox_dialogue_active
		trigger_event_non_silently = { id = reformation.21 days = { 180 365 } }
		religion:lutheran = {
			set_mutual_religious_view = {
				target = religion:orthodox
				value = positive
			}
		}
		scope:patriarch_country = {
			if = {
				limit = { has_societal_value = societal_value_type:latinization_vs_hellenization }
				change_societal_value = { type = latinization_vs_hellenization value = societal_value_minor_move_to_left }
			}
		}
	}
}

# An Orthodox Scholar Arrives
# Fires at the Lutheran country some months after the positive outcome
reformation.21 = {
	type = country_event
	category = situation_event
	title = reformation.21.title
	desc = reformation.21.desc
	outcome = neutral
	historical_info = reformation.21.historical_info

	trigger = {
		current_date > 476.9.4
		has_variable = lutheran_orthodox_dialogue_active
	}

	illustration_tags = {
		10 = regular
		10 = interior
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
		create_character = {
			religion = religion:orthodox
			estate = estate_type:clergy_estate
			adm = { 40 60 }
			dip = { 40 60 }
			mil = { 0 20 }
			create_in_limbo = yes
			save_scope_as = orthodox_scholar
		}
		remove_variable = lutheran_orthodox_dialogue_active
	}

	option = {
		name = reformation.21.a
		add_prestige = prestige_mild_bonus
		scope:orthodox_scholar = { move_country = root }
	}

	option = {
		name = reformation.21.b
		add_religious_influence = religious_influence_weak_bonus
	}
}
