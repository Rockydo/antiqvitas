		save_scope_as = target_country

	}

	option = {

		if = {
			limit = {
				any_in_global_list = {
					variable = iw_italian_leagues_list
					any_international_organization_member = {
						this != root

							has_truce_with = root
							root = { has_truce_with = prev }
						}
					}
				}
			}
			every_in_global_list = {
				variable = iw_italian_leagues_list
				every_international_organization_member = {
					limit = {
						this != root

							has_truce_with = root
							root = { has_truce_with = prev }
						}
					}
					remove_truce_with = root
					hidden_effect = {
						root = { remove_truce_with = prev }
					}
				}
			}
		}
		if = {
			limit = {
				any_in_global_list = {
					variable = iw_foreign_leagues_list
					any_international_organization_member = {
						this != root
						OR = {
							has_truce_with = root
							root = { has_truce_with = prev }
						}
					}
				}
			}
			every_in_global_list = {
				variable = iw_foreign_leagues_list
				every_international_organization_member = {
					limit = {
						this != root
						OR = {
							has_truce_with = root
							root = { has_truce_with = prev }
						}
					}
					remove_truce_with = root
					hidden_effect = {
						root = { remove_truce_with = prev }
					}
				}
			}
		}
	
		# TODO (separate MR): implement the actual mechanic that allows foreign league leaders to call in members for offensive wars at 90+ tension

		custom_tooltip = all_leaders_of_leagues_annul_all_truces_tt
	}
	trigger = {

	}

italian_wars.100 = {
	hide_portraits = yes
	type = country_event
	category = situation_event	

	title = italian_wars.100.title
	desc = italian_wars.100.desc
	outcome = neutral

	image = "gfx/interface/illustrations/situation/italian_wars.dds"

	immediate = {
		random_international_organizations_member_of = {
			limit = {
				OR = {
					this = international_organization:italian_league_1
					this = international_organization:italian_league_2
					this = international_organization:italian_league_3
				}
			}
			save_scope_as = target_italian_io
			leader_country = {
				save_scope_as = target_country
				ruler_or_regent ?= { save_scope_as = target_character }
			}
		}

		if = {
			limit = {
				has_special_status_in_international_organization = {
					type = special_status:italian_league_sponsor
					international_organization = scope:target_italian_io
				}
			}

			set_variable = {
				name = was_sponsor_variable
				years = 1
			}
		}
	}
	
	option = {
		name = italian_wars.100.a

		ai_will_select = {
			add = 100
			add = { value = "opinion(scope:target_country)" }
		}
		add_prestige = prestige_severe_bonus
		add_stability = stability_mild_bonus
		if = {
			limit = {
				scope:target_italian_io = {
					any_international_organization_member = {
						is_subject = no
						count > 1
					}
				}
				scope:target_country = root
			}
			custom_tooltip = demand_vassalization_from_members_tt
		}
		if = {
			limit = { scope:target_country = root }
			add_country_modifier = {
				modifier = iw_victory_in_the_italian_wars_modifier
				mode = replace
				years = -1
			}
			scope:target_character = {
				add_adm = { 10 25 }
				add_dip = { 10 25 }
				add_mil = { 10 25 }
			}
			unlock_law_effect = { type = administration_of_italian_lands_law }
		}
		else_if = {
			limit = {
				is_subject = no
				capital.region = region:italy_region
				NOT = { has_variable = was_sponsor_variable }
			}
			make_subject_of = { target = scope:target_country type = subject_type:vassal }
		}
	}

	option = {
		name = italian_wars.100.b

		ai_will_select = {
			add = 100
			add = {
				value = "opinion(scope:target_country)"
				multiply = -1
			}
		}

		trigger = {
			scope:target_country != root
			is_subject = no
			NOT = { has_variable = was_sponsor_variable }
		}

		scope:target_country = {
			add_casus_belli = {
				target = root
				type = casus_belli:cb_iw_annexation
				years = 25
			}
		}

		if = {
			limit = { is_allied_with = { target = scope:target_country } }
			remove_relation = {
				type = relation_type:alliance
				first = this
				second = scope:target_country
			}
		}
	}
	trigger = {
		current_date > 476.9.4
	}
}

italian_wars.300 = {	
	type = country_event
	category = situation_event	

	title = italian_wars.300.title
	desc = italian_wars.300.desc
	outcome = neutral
	image = "gfx/interface/illustrations/situation/italian_wars.dds"
	immediate = {
		random_international_organizations_member_of = {
			limit = {
				OR = {
					this = international_organization:foreign_league_balkan
					this = international_organization:foreign_league_france
					this = international_organization:foreign_league_hre
					this = international_organization:foreign_league_iberia
				}
			}
			save_scope_as = target_io
			leader_country = {
				save_scope_as = target_country
				ruler_or_regent ?= { save_scope_as = target_character }
			}
		}

		if = {
			limit = { root = scope:target_country }
			set_variable = {
				name = io_location_scale_variable
				value = scope:target_io.total_locations_owned
			}
		}
	}
	
	option = {
		name = italian_wars.300.a
		if = {
			limit = {
				root != scope:target_country
			}
			add_opinion_mutual_effect = {
				target = scope:target_country
				modifier = fought_with_us_modifier
			}
		}
		add_country_modifier = {
			modifier = iw_victory_in_the_italian_wars_modifier
			years = -1
			mode = add_and_extend
		}
		if = {
			limit = {
				root = scope:target_country
			}
			add_prestige = prestige_extreme_bonus
			# France unique reward — scaled by IO locations
			if = {
				limit = { scope:target_io = international_organization:foreign_league_france }
				add_country_modifier = {
					modifier = iw_france_victory_modifier
					years = modifier_duration_years_long
					mode = replace
					size = {
						add = {
							value = 1
							desc = BASE_VALUE
						}
						add = {
							value = var:io_location_scale_variable
							desc = "league_locations_owned_desc"
						}
					}
				}
			}
			# Iberia unique reward — scaled by IO locations
			else_if = {
				limit = { scope:target_io = international_organization:foreign_league_iberia }
				add_country_modifier = {
					modifier = iw_iberia_victory_modifier
					years = modifier_duration_years_long
					mode = replace
					size = {
						add = {
							value = 1
							desc = BASE_VALUE
						}
						add = {
							value = var:io_location_scale_variable
							desc = "league_locations_owned_desc"
						}
					}
				}
			}
			# Balkan unique reward — tiered by tag and DLC
			else_if = {
				limit = { scope:target_io = international_organization:foreign_league_balkan }
				if = {
					limit = {
						has_dlc = "d008_fate_of_the_phoenix"
						OR = { tag = ROM  tag = BYZ }
					}
					if = {
						limit = {
							OR = {
								tag = ROM
								societal_value:latinization_vs_hellenization > 0
							}
						}
						add_country_modifier = {
							modifier = iw_rom_byz_latin_victory_dlc_modifier
							years = -1
							mode = replace
						}
					}
					else = {
						add_country_modifier = {
							modifier = iw_byz_hellenic_victory_dlc_modifier
							years = -1
							mode = replace
						}
					}
				}
				else_if = {
					limit = { OR = { tag = ROM  tag = BYZ } }
					add_country_modifier = {
						modifier = iw_balkan_victory_no_dlc_modifier
						years = -1
						mode = replace
					}
				}
				else_if = {
					limit = { country_rank != country_rank:rank_empire }
					set_country_rank_effect = { rank = country_rank:rank_empire }
				}
				else = {
					add_country_modifier = {
						modifier = iw_balkan_victory_cabinet_modifier
						years = -1
						mode = replace
					}
				}
			}
			unlock_law_effect = { type = administration_of_italian_lands_law }
		}
		else = {
			add_prestige = prestige_mild_bonus
		}

		if = {
			limit = {
				exists = international_organization:hre
				scope:target_io = international_organization:foreign_league_hre
				root = scope:target_country
			}
			change_imperial_authority = { value = imperial_authority_unique_bonus }
			custom_tooltip = non_princes_become_princes_tt
		}
		else_if = {
			limit = {
				root != scope:target_country
				NOT = { is_member_of_international_organization = international_organization:hre }
				exists = international_organization:hre
			}
			scope:target_io = { add_country_to_international_organization = root }
		}
	}

	after = {
		if = {
			limit = { root = scope:target_country }
			hidden_effect = { remove_variable = io_location_scale_variable }
		}
	}
	trigger = {
		current_date > 476.9.4
	}
}

italian_wars.20 = { #The Condottiere's Price
	type = country_event
	category = situation_event

	title = italian_wars.20.title
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					exists = scope:target_character
					scope:target_character = {
						is_ruler = no
						is_regent = no
						is_consort = no
						is_heir = no
					}
				}
				desc = italian_wars.20.desc.a
			}
			triggered_desc = {
				trigger = { NOT = { exists = scope:target_character } }
				desc = italian_wars.20.desc.b
			}
			triggered_desc = {
				trigger = { always = yes }
				desc = italian_wars.20.desc.c
			}
		}
	}
	outcome = neutral

	image = "gfx/interface/illustrations/situation/italian_wars.dds"

	trigger = {
		current_date > 476.9.4
		is_member_of_any_italian_wars_league = yes
		NOT = { has_variable = recently_had_iw20_variable }
		at_war = no
		any_army = {
			any_sub_unit = {
				is_mercenary = yes
			}

			unit_location = {
				top_owner ?= ROOT
			}
		}
	}

	immediate = {

		random_army = {
			limit = {
				any_sub_unit = {
					is_mercenary = yes
				}
				unit_location = {
					top_owner ?= ROOT
				}
			}
			if = {
				limit = { 
					has_commander = yes
					leader = {
						is_ruler = no
						is_regent = no
						is_consort = no
						is_heir = no
					}
				}
				leader = { save_scope_as = target_character }
				
			}
			save_scope_as = target_merc
			unit_location = { 
				save_scope_as = target_location 
				province = { save_scope_as = target_province }
			}
		}

		set_variable = {
			name = recently_had_iw20_variable
			years = 10
		}
	}

	option = {	#Pay them off
		name = italian_wars.20.a

		ai_will_select = {
			add = {
				value = gold
				divide = monthly_income_total
				multiply = 10
			}
		}

		change_gold_effect = { scale = -2 }

		custom_tooltip = mercs_among_the_ranks_of_target_merc_are_reigned_in_tt
	}

	option = {
		name = italian_wars.20.b

		ai_will_select = {
			value = 60
			subtract = {
				value = gold
				divide = monthly_income_total
				multiply = 10
			}
		}

		scope:target_province = {
			change_province_food_percentage = -0.5
			custom_tooltip = {
				text = target_province_loses_prosperity_extreme_penalty_tt
				every_location_in_province = { change_prosperity = prosperity_extreme_penalty }
			}
		}
	}

	option = {	#Ruler personally renegotiates — half cost, prestige gain
		name = italian_wars.20.c

		trigger = {
			exists = ruler_or_regent
			ruler_or_regent = {
				OR = {
					has_trait = silver_tongue
					has_trait = careful
					total_abilities >= 220
				}
			}
		}

		ai_will_select = {
			value = 200
		}

		change_gold_effect = { scale = -1 }
		add_prestige = prestige_mild_bonus

		custom_tooltip = mercs_among_the_ranks_of_target_merc_are_reigned_in_tt
	}
}

italian_wars.21 = { #Battle-Hardened Veterans
	type = country_event
	category = situation_event

	title = italian_wars.21.title
	desc = italian_wars.21.desc
	outcome = positive

	image = "gfx/interface/illustrations/situation/italian_wars.dds"

	trigger = {
		current_date > 476.9.4
		is_member_of_any_italian_wars_league = yes
		NOT = { has_variable = recently_had_iw21_variable }
		at_war = no
		any_army = {
			experience_percentage > 0.25
			experience_percentage < 1
		}
	}

	immediate = {
		set_variable = {
			name = recently_had_iw21_variable
			years = 15
		}

		ordered_army = {
			limit = {

			value = 20

				value = stability

		}

			value = 200
		}

		scope:target_army = {
			add_unit_modifier = {
				modifier = battle_hardened_veterans_modifier

				mode = replace
			}
		}

			type = belligerent_vs_conciliatory
			value = societal_value_move_to_left
		}
	}

italian_wars.22 = { #Growing Distaste to War
	type = country_event
	category = situation_event

	title = italian_wars.22.title
	desc = italian_wars.22.desc
	outcome = negative

	image = "gfx/interface/illustrations/situation/italian_wars.dds"

	trigger = {
		current_date > 476.9.4
		is_member_of_any_italian_wars_league = yes
		NOT = { has_variable = recently_had_iw22_variable }
		at_war = no
		regular_army_size > 0
		monthly_manpower > 0
	}

	immediate = {
		set_variable = {
			name = recently_had_iw22_variable
			years = 15
		}

		random_estate = {
			limit = {
				NOT = { estate_type = estate_type:crown_estate }
			}
			save_scope_as = hippie_estate
		}

		random_list = {
			1 = { set_variable = iw22_variant_a1 }
			1 = { set_variable = iw22_variant_a2 }
			1 = { set_variable = iw22_variant_a3 }
		}
		random_list = {
			1 = { set_variable = iw22_variant_b1 }
			1 = { set_variable = iw22_variant_b2 }
			1 = { set_variable = iw22_variant_b3 }
		}
	}

	option = {	#Cowards all of you!
		name = italian_wars.22.a

		switch = {
			trigger = has_variable
			iw22_variant_a1 = {
				add_yearly_manpower = -2
				if = {
					limit = { exists = scope:hippie_estate }
					add_estate_satisfaction = {
						type = scope:hippie_estate.estate_type
						value = estate_satisfaction_mild_bonus
					}
				}
			}
			iw22_variant_a2 = {
				add_yearly_manpower = -2
				add_army_tradition = army_tradition_extreme_penalty
				add_estate_satisfaction = {
					type = scope:hippie_estate.estate_type
					value = estate_satisfaction_mild_penalty
				}
			}
			iw22_variant_a3 = {
				add_legitimacy = legitimacy_severe_penalty
				add_estate_satisfaction = {
					type = scope:hippie_estate.estate_type
					value = estate_satisfaction_mild_penalty
				}
			}
		}
	}

	option = {	#Not one step back!
		name = italian_wars.22.b

		switch = {
			trigger = has_variable
			iw22_variant_b1 = {
				add_war_exhaustion = war_exhaustion_extreme_penalty
				add_estate_satisfaction = {
					type = scope:hippie_estate.estate_type
					value = estate_satisfaction_extreme_penalty
				}
			}
			iw22_variant_b2 = {
				add_stability = stability_severe_penalty
				add_estate_satisfaction = {
					type = scope:hippie_estate.estate_type
					value = estate_satisfaction_extreme_penalty
				}
			}
			iw22_variant_b3 = {
				iw_increase_tension_effect = { value = iw_tension_gain_low }
				add_estate_satisfaction = {
					type = scope:hippie_estate.estate_type
					value = estate_satisfaction_extreme_penalty
				}
			}
		}
	}

	after = {
		hidden_effect = {
			remove_variable = iw22_variant_a1
			remove_variable = iw22_variant_a2
			remove_variable = iw22_variant_a3
			remove_variable = iw22_variant_b1
			remove_variable = iw22_variant_b2
			remove_variable = iw22_variant_b3
		}
	}
}

italian_wars.23 = { #The Turncoat General
	type = country_event
	category = situation_event

	title = italian_wars.23.title
	desc = italian_wars.23.desc
	outcome = negative

	image = "gfx/interface/illustrations/situation/italian_wars.dds"

	trigger = {
		current_date > 476.9.4
		is_member_of_any_italian_wars_league = yes
		NOT = { has_variable = recently_had_iw23_variable }

		any_army = { 
			has_commander = yes
			leader = {
				is_ruler = no
				is_regent = no
				is_consort = no
				is_heir = no
			}
		}
	}

	major = yes
	major_trigger = {
		this = scope:target_country_recipient
	}

	immediate = {
		set_variable = {
			name = recently_had_iw23_variable
			years = 10
		}

		ruler_or_regent ?= { save_scope_as = target_character1 }

		random_army = {
			limit = {
				has_commander = yes
				leader = {
					is_ruler = no
					is_regent = no
					is_consort = no
					is_heir = no
				}
			}
			save_scope_as = target_army
			leader = { save_scope_as = target_character2 }
		}
		every_in_global_list = {
			variable = iw_italian_leagues_list
			if = {
				limit = { country_exists = leader_country }
				leader_country = { add_to_list = iw23_list }
			}
		}
		every_in_global_list = {
			variable = iw_foreign_leagues_list
			if = {
				limit = { country_exists = leader_country }
				leader_country = { add_to_list = iw23_list }
			}
		}

		if = {
			limit = {
				any_in_list = {
					list = iw23_list
					is_at_war_with = root
				}
			}
			random_in_list = {
				list = iw23_list
				limit = { is_at_war_with = root }
				save_scope_as = target_country_recipient
			}
		}
		else = {
			random_in_list = {
				list = iw23_list
				save_scope_as = target_country_recipient
			}
		}
	}

	option = {	#Good riddance turncoat!
		name = italian_wars.23.a

		ai_will_select = {
			value = 40
			subtract = {
				value = gold
				divide = monthly_income_total
				multiply = 5
			}
		}

		scope:target_character2 = { move_country = scope:target_country_recipient }
		add_prestige = prestige_mild_penalty
	}

	option = {	#Buy [target_character2.GetHerHis] allegiance!
		name = italian_wars.23.b

		ai_will_select = {
			value = 20
			add = {
				value = gold
				divide = monthly_income_total
				multiply = 5
			}
			add = army_tradition
			add = {
				value = scope:target_character2.total_abilities
				subtract = 150
			}
		}

		change_gold_effect = { scale = -6 }
		add_prestige = prestige_mild_penalty
	}

	option = {	#Ruler confronts the traitor personally
		name = italian_wars.23.c

		trigger = {
			exists = ruler_or_regent
			ruler_or_regent = {
				OR = {
					has_trait = inspiring_leader
					has_trait = strict
					total_abilities >= 220
				}
			}
		}

		ai_will_select = {
			value = 200
		}

		custom_tooltip = target_character2_wont_attempt_to_flee_for_10_years_tt
	}
}

			OR = {

				international_organization_type = international_organization_type:foreign_league_hre
			}

		}
	}

	immediate = {
		set_variable = {
			name = recently_had_iw24_variable
			years = 4
		}

		random_international_organizations_member_of = {
			limit = {
				OR = {
					this = international_organization:italian_league_1
					this = international_organization:italian_league_2
					this = international_organization:italian_league_3
					this = international_organization:foreign_league_balkan
					this = international_organization:foreign_league_france
					this = international_organization:foreign_league_iberia
					this = international_organization:foreign_league_hre
				}
			}
			save_scope_as = target_io
		}

		scope:target_io = {
			random_international_organization_member = {
				limit = { this != root }
				save_scope_as = target_country
			}
		}
	}

	option = {
		name = italian_wars.24.a 
		trigger = {
			exists = ruler_or_regent
			ruler_or_regent = {
				OR = {
					has_trait = calm
					has_trait = well_advised
					total_abilities >= 220
				}
			}
		}

		custom_tooltip = ruler_or_regent_skillful_diplomacy_saves_the_day
	}
	option = {
		name = italian_wars.24.b

		if = {
			limit = {
				scope:target_country = {
					has_opinion = { modifier = opinion_diplomatic_accord target = root }
				}
			}
			remove_opinion = { modifier = opinion_diplomatic_accord target = scope:target_country }
			scope:target_country = { remove_opinion = { modifier = opinion_diplomatic_accord target = root } }
		}
		if = {
			limit = {
				scope:target_country = {
					NOT = { has_opinion = { modifier = opinion_diplomatic_misunderstanding target = root } }
				}
			}
			scope:target_country = {
				add_opinion_mutual_effect = {
					modifier = opinion_diplomatic_misunderstanding
					target = root
				}
			}
		}
		else = { add_government_power = government_power_mild_penalty }
	}
}

italian_wars.26 = { #Diplomatic Incident
	type = country_event
	category = situation_event

	title = italian_wars.26.title
	desc = italian_wars.26.desc
	outcome = negative

	image = "gfx/interface/illustrations/situation/italian_wars.dds"

	trigger = {
		current_date > 476.9.4
		at_war = no
		NOT = { has_variable = recently_had_iw26_variable }
		OR = {
			is_leader_of_international_organization = international_organization:italian_league_1
			is_leader_of_international_organization = international_organization:italian_league_2
			is_leader_of_international_organization = international_organization:italian_league_3
			is_leader_of_international_organization = international_organization:foreign_league_balkan
			is_leader_of_international_organization = international_organization:foreign_league_france
			is_leader_of_international_organization = international_organization:foreign_league_iberia
			is_leader_of_international_organization = international_organization:foreign_league_hre
		}
		any_neighbor_country = {
			capital.region = region:italy_region
			root = { 
				NOT = { has_truce_with = prev }
			}
			is_member_of_any_italian_wars_league = yes
			NOT = { is_subject_of = root }
			root = { can_declare_war_on = prev }
			relative_military_strength = {
				target = root
				value < 0.4
			} 
		}
	}

	immediate = {
		set_variable = {
			name = recently_had_iw26_variable
			years = 10
		}

		random_neighbor_country = {
			limit = {
				capital.region = region:italy_region
				root = { 
					NOT = { has_truce_with = prev }
				}
				is_member_of_any_italian_wars_league = yes
				NOT = { is_subject_of = root }
				root = { can_declare_war_on = prev }
				relative_military_strength = {
					target = root
					value < 0.4
				} 
			}
			save_scope_as = target_country
			ruler_or_regent ?= { save_scope_as = target_character }
		}

		random_international_organizations_member_of = {
			limit = {
				leader_country = root
				OR = {
					this = international_organization:italian_league_1
					this = international_organization:italian_league_2
					this = international_organization:italian_league_3
					this = international_organization:foreign_league_balkan
					this = international_organization:foreign_league_france
					this = international_organization:foreign_league_iberia
					this = international_organization:foreign_league_hre
				}
			}
			save_scope_as = our_io
		}
	}

	option = {
		name = italian_wars.26.a

		ai_will_select = {
			if = {
				limit = { scope:target_country = { is_subject = yes } }
				add = {
					value = 100
					multiply = "relative_strength(scope:target_country.overlord)"
					divide = 5
					desc = "RELATIVE_STRENGTH"
				}
			}
			else = {
				add = {
					value = 100
					multiply = "relative_strength(scope:target_country)"
					divide = 5
					desc = "RELATIVE_STRENGTH"
				}
