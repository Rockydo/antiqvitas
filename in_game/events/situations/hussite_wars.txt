hussite_wars.1 = { #Jan Želivský's Congregation
	type = country_event
	category = situation_event

		first_valid = {
			triggered_desc = {
				trigger = {
					OR = {
						religion = religion:catholic
						religion = religion:hussite
					}
				}
				desc = hussite_wars.1.desc
			}
			triggered_desc = { 
				desc = hussite_wars.1.desc.other_heresy
			}
		}
	}
	outcome = neutral
	image = "gfx/interface/illustrations/situation/hussite_wars.dds"
	dynamic_historical_event = {
		tag = BOH
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 3
	}

	trigger = {
		current_date > 476.9.4
		religion = religion:catholic
		is_subject = no
		character:boh_jan_hus = {
			is_alive = no
		}
		country_exists = international_organization:hre.leader_country
		NOT = { is_leader_of_international_organization = international_organization:hre }
	}

	fire_only_once = yes
	immediate = {
		create_character = {
			first_name = name_john
			last_name = zelivsky
			adm = { 15 100 }
			dip = { 50 100 }
			mil = { 10 100 }
			birth_date = 476.9.4
			culture = culture:czech
			religion = religion:hussite
			estate = estate_type:clergy_estate
			save_scope_as = jan_zelivsky_scope
		}
		capital = { save_scope_as = target_boh_capital_scope }
		international_organization:hre.leader_country ?= { save_scope_as = hre_emperor_scope }
		ordered_owned_location = {
			order_by = {
				value = population
				multiply = "religion_percentage(religion:hussite)"
			}
			max = 1
			check_range_bounds = no
			save_scope_as = taborite_capital_scope
		}
		create_rebel = {
			category = estate
			estate = clergy_estate
			name = hussite_rebellion
			save_scope_as = hussite_rebel_scope
		}
		lots_of_religion = {
			religion = religion:hussite
			other_religion = religion:catholic
		}
	}

	option = {
		name = hussite_wars.1.a
		ai_will_select = { value = 95 }	
		historical_option = yes

		if = {
			limit = {
				NOT = { religion = religion:hussite }
			}
			change_religion = religion:hussite
			change_religion_for_ruler_and_family = { country = ROOT religion = religion:hussite }
		}
		else = { add_religious_influence = religious_influence_mild_bonus }

		hidden_effect = {
			if = {
				limit = {
					exists = international_organization:hre
					international_organization:hre = {
						international_organization_has_leader = yes
						leader_country ?= {
							ruler ?= {
								NOT = { hre_country_fulfills_imperial_religion_requirement = { type = voting } }
							}
						}
					}
				}
				international_organization:hre = {
					leader_country ?= { save_scope_as = heretic_emperor_scope }
					international_organization_remove_special_status = {
						type = special_status:emperor
						country = scope:heretic_emperor_scope
					}
				}
				international_organization_chooses_new_leader = international_organization:hre
			}
		}

		if = {
			limit = {
				has_ruler = yes
				ruler = { has_character_modifier = enacted_placitum_regium }
			}

		}

		upset_countries_of_religion = {
			religion = religion:catholic
			modifier = religious_disagreements_modifier
		}

		hidden_effect = { destroy_rebel = scope:hussite_rebel_scope }
		add_religious_influence = religious_influence_ultimate_bonus
		add_country_modifier = {
			modifier = in_defense_of_hussitism_modifier
			years = modifier_duration_years_short
			mode = replace
		}
	}

	option = {
		name = hussite_wars.1.b
		ai_will_select = { value = 5 }	

		custom_tooltip = end_hussite_wars_situation_tt
		if = {
			limit = {
				NOT = { religion = religion:catholic }
			}
			change_religion = religion:catholic
			change_religion_for_ruler_and_family = { country = ROOT religion = religion:catholic }
		}

		custom_tooltip = {
			text = hussite_rebellion_tooltip
			every_pop = {
				limit = {
					owner = root
					religion = religion:hussite
				}
				add_pop_satisfaction = pop_satisfaction_radical_penalty
				change_pop_allegiance = scope:hussite_rebel_scope
			}
		}

		add_estate_satisfaction = { 
			type = estate_type:clergy_estate
			value = estate_satisfaction_radical_penalty
		}
		add_stability = stability_ultimate_penalty
		add_government_power = government_power_radical_penalty

		add_country_modifier = {
			modifier = civil_war_looming
			years = modifier_duration_years_normal
			mode = add_and_extend
		}
	}

	after = {
		show_as_tooltip = {
			lots_of_religion = {
				religion = religion:hussite
				other_religion = religion:catholic
			}
		}
	}
}
