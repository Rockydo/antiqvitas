namespace = movements

#(root is the country, scope:movement_definition is the movement)
movements.10 = {
	type = country_event
	category = situation_event	

	title = movements.10.title
	desc = movements.10.desc
	outcome = neutral

	illustration_tags = {
		10 = happy
		10 = exterior
	}

	trigger = {
		current_date > 476.9.4
		OR = {
			has_or_had_tag = BYZ
			has_or_had_tag = ROM
		}
	}

	immediate = {
		event_illustration_culture_effect = { culture = culture:roman_culture }
	}

	option = {
		name = movements.10.a
		add_accepted_culture = culture:roman_culture
		custom_tooltip = movements.10.tt
		
		ordered_owned_location = {
			limit = { 
				roman_culture_movement_location_trigger = yes
				NOT = { any_pop = { culture = culture:roman_culture } }
			}
			max = 5

			order_by = {
				value = population

				if = {
					limit = { location_rank = location_rank:megalopolis }
					add = 100000
				}
				else_if = {
					limit = { location_rank = location_rank:city }
					add = 10000
				}

				multiply = local_control
				multiply = average_satisfaction
			}
			save_scope_as = new_target_location
			culture:roman_culture = {
				random_movement_in_culture = {
					spread_to_location = {
						location = scope:new_target_location
						supporters_percent = {
							value = 0.2
							multiply = scope:new_target_location.local_control
							multiply =scope:new_target_location.average_satisfaction
							min = 0.05
						}
					}
				}
			}
		}
	}
	option = {
		name = movements.10.b
		if = {
			limit = { culture = culture:greek_culture }
			change_culture = culture:roman_culture
			add_accepted_culture = culture:greek_culture
		}
		else = {
			change_culture = culture:roman_culture
		}
		custom_tooltip = movements.10.tt
		
		ordered_owned_location = {
			limit = { 
				roman_culture_movement_location_trigger = yes
				NOT = { any_pop = { culture = culture:roman_culture } }
			}
			max = 5

			order_by = {
				value = population

				if = {
					limit = { location_rank = location_rank:megalopolis }
					add = 100000
				}
				else_if = {
					limit = { location_rank = location_rank:city }
					add = 10000
				}

				multiply = local_control
				multiply = average_satisfaction
			}

			save_scope_as = new_target_location
			culture:roman_culture = {
				random_movement_in_culture = {
					spread_to_location = {
						location = scope:new_target_location
						supporters_percent = {
							value = 0.2
							multiply = scope:new_target_location.local_control
							multiply =scope:new_target_location.average_satisfaction
							min = 0.05
						}
					}
				}
			}
		}
	}
}


#(root is the country, scope:movement_definition is the movement)
movements.20 = {
	type = country_event
	category = situation_event

	title = movements.20.title
	desc = movements.20.desc
	outcome = neutral

	illustration_tags = {
		10 = regular
		10 = interior
	}

	immediate = {
		event_illustration_religion_effect = { religion = religion:hellenism_religion }
	}
	

	trigger = {
		current_date > 476.9.4
		OR = {
			culture = { has_culture_group = culture_group:greek_group }
			culture = culture:roman_culture
			any_owned_location = {
				any_pop = {
					OR = {
						culture = { language = language:greek_language }
						culture = { language ?= { language_family ?= { this = language_family:romance_language_family } } }
					}
				}
			}
		}
	}

	option = {
		name = movements.20.a
		custom_tooltip = movements.20.tt
		religion:hellenism_religion = {
			random_movement_in_religion = {
				spread_to_location = {
					location = root.capital
					supporters_percent = {
						min = 0.05
					}
				}
			}
		}
	}

	option = {
		name = movements.20.b
		add_country_modifier = {
			modifier = hellenism_religion_movement_suppressed
			months = 60
		}
	}
}
