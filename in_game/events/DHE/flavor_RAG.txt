namespace = flavor_rag

flavor_rag.1000 = { #Enemy at the Gates
	type = country_event
	title = flavor_rag.1000.title
	desc = flavor_rag.1000.desc
	outcome = neutral

	illustration_tags = {
		10 = interior
		10 = regular
	}

	fire_only_once = yes
	
	historical_info = flavor_rag.1000.historical_info

	dynamic_historical_event = {
		tag = RAG
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 1
	}
	
	immediate = {
		ruler ?= {
			save_scope_as = target_ruler
		}
		c:TUR = {
			ruler ?= {
				save_scope_as = ottoman_ruler
			}
		}
	}

	trigger = {
		current_date > 476.9.4
		is_subject = no
		country_exists = c:TUR
		any_neighbor_country = {
			tag = TUR
		}
		c:TUR = {
			NOT = { is_allied_with = { target = root } }
			is_great_power = yes
			country_economical_base > c:RAG.country_economical_base
		}
	}
	
	option = {
		name = flavor_rag.1000.a
		historical_option = yes
		add_prestige = prestige_severe_penalty
		change_societal_value = {
			type = belligerent_vs_conciliatory
			value = societal_value_minor_move_to_right
		}
		c:TUR = {
			trigger_event_non_silently = flavor_rag.1001
		}
	}
	
	option = {
		name = flavor_rag.1000.b
		add_prestige = prestige_mild_bonus
		change_societal_value = {
			type = belligerent_vs_conciliatory
			value = societal_value_minor_move_to_left
		}
	}
}

	}

			target = c:RAG

		}
	}
	trigger = {
		current_date > 476.9.4
	}
}
