scripted_effect set_up_della_scala_targets = {
	random_war = {
		limit = {
			casus_belli ?= casus_belli:cb_scaligeri_war
		}
		save_scope_as = target_war
	}
	dynasty:scala_dynasty = {
		save_scope_as = target_dynasty
	}
}

# Scaligeri War - Verona
flavor_ver.1 = {
	type = country_event
	title = flavor_ver.1.title
	desc = flavor_ver.1.desc
	historical_info = flavor_ver.1.historical_info
	outcome = neutral

	immediate = {
		set_up_della_scala_targets = yes
		event_illustration_estate_effect = {
			background = estate_type:burghers_estate
			foreground = estate_type:nobles_estate
		}
		
		scope:target_war = {
			every_attacker = {
				limit = {
					is_subject = no
				}
				add_to_list = anti_scaligeri_war_participants
			}
		}
		save_scope_as = target_ver
		c:VEN = {
			save_scope_as = anti_scaligeri_leader
			trigger_event_non_silently = {
				id = flavor_ver.2
			}
			trigger_event_non_silently = {
				id = flavor_ver.1000
				days = { 30 60 }
			}
		}
		
		character:ver_cangrande_scala = {
			save_scope_as = target_character
		}
	}

	illustration_tags = {
		10 = armed
		10 = exterior
	}

	option = {
		name = flavor_ver.1.a
		
		custom_tooltip = VER_benefits_from_winning_the_war
	}
	trigger = {
		current_date > 476.9.4
	}
}

scripted_effect scaligeri_war_demand_antagonism_high = {

		target = root.capital

# scope: none country: string flag: string
scripted_effect scaligeri_war_peace_treaty_choice = {
	custom_tooltip = {
		text = ver_demands_$country$_$flag$
		scope:target_war = {
			set_variable = {
				name = $country$_demands
				value = flag:$flag$
			}
		}
	}
	scaligeri_war_demand_antagonism_$antagonism$ = yes
}

scripted_trigger root_or_scaligeri_war_participant_controls = {
	$location$ = {
		custom_tooltip = {
			text = ver_owner_is_an_defender
			owner ?= {
				scope:target_war = {
					is_a_defender = PREV
				}
			}
		}
		custom_tooltip = {
			text = ver_controller_is_an_attacker
			controller ?= {
				scope:target_war = {
					is_an_attacker = PREV 
				}
			}
		}
	}
}

	dynamic_historical_event = {
		tag = MLO
		from = 476.9.4
		to = 476.9.4

		current_date > 476.9.4

			save_scope_as = war_leader

			save_scope_as = target_country

				save_scope_as = target_province
			}
		}

			save_scope_as = target_peace_treaty
		}
	}

		10 = armed
		10 = exterior
	}

		name = flavor_ver.3.b

flavor_ver.1003 = {
	type = country_event
	title = flavor_ver.1003.title
	desc = flavor_ver.1003.desc
	outcome = positive

	trigger = {
		current_date > 476.9.4
		padua_incident_is_valid = yes
		any_current_war = {
			casus_belli ?= casus_belli:cb_scaligeri_war
			is_an_attacker = ROOT
		}
		scope:target = location:padova
		scope:target.owner ?= {
			is_at_war_with = ROOT
		}
	}

	illustration_tags = {
		10 = armed
		10 = exterior
	}

	immediate = {
		set_up_della_scala_targets = yes

		event_illustration_estate_effect = {
			background = estate_type:burghers_estate
			foreground = estate_type:nobles_estate
		}
		
		location:padova = {
			save_scope_as = target_location
		}
		c:PAD = {
			save_scope_as = target_country
		}
		estate_type:burghers_estate = {
			save_scope_as = target_estate
		}
		character:pad_marsiglio_carrara = {
			save_scope_as = target_character
		}
		c:PAD.overlord = {
			trigger_event_non_silently = {
				id = flavor_ver.1005
			}
		}

		international_organization:guelphs_io ?= {
			save_scope_as = guelphs_io
		}
	}

	option = {
		name = flavor_ver.1003.a
		
		scope:target_country = {
			overlord ?= {
				cancel_subject = scope:target_country
			}
			set_new_ruler = scope:target_character
			leave_war = {
				war = scope:target_war 
				actor = scope:target_country
			}
			custom_tooltip = {
				text = ver_will_join_target_war
				join_war_as_attacker = {
					war = scope:target_war
				}
			}
		}
		if = {
			limit = {
				exists = scope:guelphs_io
				scope:target_country = {
					NOT = {
						is_member_of_international_organization_of_type = { type = guelphs_io }
					}
				}
			}
			join_situation_faction = {
				target = scope:target_country
				international_organization = scope:guelphs_io
				join_variable = gag_recently_joined_a_faction
				join_years = 5
				leave_variable = gag_recently_left_a_faction
			}
		}
	}
}
