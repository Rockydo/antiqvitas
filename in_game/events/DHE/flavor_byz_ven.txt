namespace = flavor_byz_ven

flavor_byz_ven.1 = {
	type = country_event
	title = flavor_byz_ven.1.title
	desc = flavor_byz_ven.1.desc
	outcome = neutral

	image = "gfx/interface/illustrations/disaster/byzantine_succession_crisis.dds"

	fire_only_once = yes
	dynamic_historical_event = {
		tag = BYZ
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 10
	}

	trigger = {
		current_date > 476.9.4
		has_or_had_tag = BYZ
		country_exists = c:VEN
		any_active_disaster = {
			OR = {
				disaster_type = disaster_type:byzantine_succession_crisis
				disaster_type = disaster_type:succession_crisis
			}
		}
		work_of_art:byzantine_crown_jewels = {
			owner ?= ROOT
		}
	}

	immediate = {
		c:VEN = {
			save_scope_as = target_country
		}
		save_scope_as = origin_country
		work_of_art:byzantine_crown_jewels = {
			save_scope_as = target_woa
		}

		set_local_variable = {
			name = loan_size
			value = {
				value = country_economical_base
				multiply = 4
				min = 250
				max = 500
			}
		}
	}


	option = {
		name = flavor_byz_ven.1.a
		
		scope:target_country = {
			trigger_event_non_silently = {
				id = flavor_byz_ven.2
			}
		}
		c:VEN = {
			custom_tooltip = BYZ_crown_jewels_tt
		}
	}

	option = {
		name = flavor_byz_ven.1.b
		
		add_prestige = prestige_weak_bonus
	}
}

flavor_byz_ven.2 = {
	type = country_event
	title = flavor_byz_ven.2.title
	desc = flavor_byz_ven.2.desc
	outcome = neutral

	image = "gfx/interface/illustrations/disaster/byzantine_succession_crisis.dds"

	option = {
		name = flavor_byz_ven.2.a
		
		add_gold = local_var:loan_size
		give_loan = {
			target = scope:origin_country
			amount = local_var:loan_size
			interest = 0.02
			months = 60
			set_loc_key = pawning_of_crown_jewels
		}
		set_variable = original_crown_jewels_lender
		scope:target_woa = {
			move_art_and_owner = { location = root.capital country = root }
		}

		scope:origin_country = {
			add_country_modifier = {
				mode = add
				modifier = BYZ_lacking_jewels
				years = -1
			}
			trigger_event_non_silently = flavor_byz_ven.9
		}
	}

	option = {
		name = flavor_byz_ven.2.b

		scope:origin_country = {
			add_opinion = {
				target = ROOT
				modifier = opinion_insulted
			}
			trigger_event_non_silently = flavor_byz_ven.10
		}
	}
	trigger = {
		current_date > 476.9.4
	}
}

flavor_byz_ven.3 = {
	type = country_event
	title = flavor_byz_ven.3.title
	desc = flavor_byz_ven.3.desc
	outcome = neutral

	image = "gfx/interface/icons/trade_goods/illustrations/icon_goods_jewelry.dds"

	trigger = {
		current_date > 476.9.4
		exists = work_of_art:byzantine_crown_jewels
		has_variable = original_crown_jewels_lender
		country_exists = c:BYZ
	}

	fire_only_once = yes

	immediate = {
		c:BYZ = {
			save_scope_as = origin_country
		}
		save_scope_as = target_country
		scope:origin_country = {
			random_owned_location = {
				save_scope_as = target_location
			}
		}
		scope:origin_country.capital ?= {
			save_scope_as = target_location
		}
		work_of_art:byzantine_crown_jewels = {
			save_scope_as = target_woa
		}
	}

	option = {
		name = flavor_byz_ven.3.a
		
		scope:target_woa = {
			move_art_and_owner = { location = scope:origin_country.capital country = scope:origin_country }
		}
		scope:origin_country = {
			trigger_event_non_silently = {
				id = flavor_byz_ven.4
			}
		}
	}

	option = {
		name = flavor_byz_ven.3.b 

		add_country_modifier = {
			modifier = BYZ_violated_agreement
			years = 40
		}
		scope:origin_country = {
			trigger_event_non_silently = {
				id = flavor_byz_ven.5
			}
		}
	}
}

flavor_byz_ven.15 = { #"An Opportunity against the Genoese Threat" ## Venice considers proposing a joint anti-Genoese pact to Byzantium in the wake of Vignoso's 1346 capture of Chios. Historical precursor to the 1351 treaty and the joint Venetian-Byzantine-Aragonese naval campaigns of 1351-1355.
	type = country_event
	fire_only_once = yes
	title = flavor_byz_ven.15.title
	desc = flavor_byz_ven.15.desc
	outcome = neutral

	dynamic_historical_event = {
		tag = VEN
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 15
	}

	trigger = {
		current_date > 476.9.4
		country_exists = c:GEN
		country_exists = c:BYZ
		c:BYZ = {
			has_opinion = { target = c:GEN modifier = gen_has_taken_chios_from_us }
			is_subject = no
			NOT = { is_allied_with = { target = root } }
			NOT = { has_truce_with = root }
		}
		NOT = { is_at_war_with = c:BYZ }
	}

	immediate = {
		save_scope_as = venetian_proposer
		c:BYZ = { save_scope_as = byzantine_pact_target }
	}

	illustration_tags = {
		10 = interior
		10 = regular
	}

	option = { #Historical - Send envoys to propose a joint pact
		name = flavor_byz_ven.15.a
		historical_option = yes

		scope:byzantine_pact_target = {
			trigger_event_non_silently = { id = flavor_byz_ven.16 }
		}
	}

	option = { #We will pursue our interests alone
		name = flavor_byz_ven.15.b
		add_prestige = prestige_weak_penalty
	}
}

flavor_byz_ven.16 = { #"The Venetian Overture" ## Byzantium receives Venice's proposed anti-Genoese pact. Triggered by flavor_byz_ven.15 option A.
	type = country_event
	title = flavor_byz_ven.16.title
	desc = flavor_byz_ven.16.desc
	outcome = positive

	illustration_tags = {
		10 = interior
		10 = regular
	}

	option = { #Historical - An old rival becomes a useful ally against a greater foe
		name = flavor_byz_ven.16.a
		historical_option = yes

		add_opinion = { target = scope:venetian_proposer modifier = byz_ven_anti_genoese_pact }
		add_country_modifier = { modifier = anti_genoese_pact_modifier years = 10 }
		scope:venetian_proposer = {
			add_opinion = { target = root modifier = byz_ven_anti_genoese_pact }
			add_opinion = { target = c:GEN modifier = ven_anti_genoese_pact_against_them }
			add_country_modifier = { modifier = anti_genoese_pact_modifier years = 10 }
		}
		if = {
			limit = { has_societal_value = societal_value_type:latinization_vs_hellenization }
			change_societal_value = {
				type = latinization_vs_hellenization
				value = societal_value_move_to_left
			}
		}
	}

	option = { #None of these Latins are of trust
		name = flavor_byz_ven.16.b
		scope:venetian_proposer = {
			add_opinion = { target = root modifier = byz_refused_venetian_pact }
		}
		if = {
			limit = { has_societal_value = societal_value_type:latinization_vs_hellenization }
			change_societal_value = {
				type = latinization_vs_hellenization
				value = societal_value_move_to_right
			}
		}
	}
	trigger = {
		current_date > 476.9.4
	}
}
