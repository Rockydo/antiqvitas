namespace = flavor_byz_ser

scripted_effect append_info_if_SMC_is_player = {
	if = {
		limit = {
			c:SMC = {
				is_ai = no
			}
		}
		custom_tooltip = SMC_player
	}
}

flavor_byz_ser.1 = {
	type = country_event
	title = flavor_byz_ser.1.title
	desc = flavor_byz_ser.1.desc
	outcome = neutral

	fire_only_once = yes
	dynamic_historical_event = {
		tag = SER
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 100
	}

	illustration_tags = {
		10 = exterior
	}

	trigger = {
		current_date > 476.9.4
		country_exists = c:BYZ
		country_exists = c:BUL
		character:ser_hrelja = {
			is_alive = yes
			owner ?= ROOT
		}
		OR = {
			location:strumica = { owner ?= ROOT }
			location:shtip = { owner ?= ROOT }
			location:kocani = { owner ?= ROOT }
		}
	}

	immediate = {
		event_illustration_estate_background_effect = {
			background = estate_type:nobles_estate
		}
		
		random_owned_location = {
			limit = {
				OR = {
					this = location:strumica
					this = location:shtip
					this = location:kocani
				}
			}
			save_scope_as = capital_for_strumica
		}
		location:strumica = {
			if = {
				limit = {
					owner ?= ROOT
				}
				save_scope_as = capital_for_strumica
			}
		}

		scope:capital_for_strumica = {
			create_country_from_location = {
				overlord = ROOT
				subject_type = subject_type:vassal
				define_unique_country_tag = SMC
				change_country_name = SMC
				change_country_adjective = SMC
				change_country_color = map_SMC
				change_country_flag = SMC
				set_new_ruler = character:ser_hrelja
			}
		}

		every_owned_location = {
			limit = {
				OR = {
					this = location:strumica
					this = location:shtip
					this = location:kocani
				}
			}
			change_location_owner = c:SMC
		}

		c:SMC.ruler = {
			save_scope_as = target_character
		}
		c:SER = {
			save_scope_as = target_country
		}
		c:SMC = {
			save_scope_as = origin_country
		}
		c:BYZ = {
			save_scope_as = target_country2
		}
		c:BUL = {
			save_scope_as = target_country3
		}
		c:BYZ = {
			trigger_event_non_silently = {
				id = flavor_byz_ser.2
			}
		}
		c:SMC = {
			trigger_event_non_silently = {
				id = flavor_byz_ser.3
			}
			trigger_event_non_silently = {
				id = flavor_byz_ser.4
				months = 44 # Roughly start of 1341
			}
		}
	}

	option = {
		name = flavor_byz_ser.1.b

		custom_tooltip = SER_strumica_was_released
		
		custom_tooltip = SER_threat_of_secession
		
		append_info_if_SMC_is_player = yes
	}


	option = {
		name = flavor_byz_ser.1.a
		
		custom_tooltip = SER_strumica_was_released

		custom_tooltip = SER_threat_of_secession

		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_left
		}

		scope:origin_country = {
			add_liberty_desire = liberty_desire_mild_plus
		}
	}
}

flavor_byz_ser.4 = {
	type = country_event
	title = flavor_byz_ser.4.title
	desc = flavor_byz_ser.4.desc
	outcome = neutral

	fire_only_once = yes

	illustration_tags = {
		10 = exterior
	}

	trigger = {
		current_date > 476.9.4
		country_exists = c:SER
		country_exists = c:BYZ
		is_subject_or_below_of = c:SER
	}

	immediate = {
		event_illustration_estate_background_effect = {
			background = estate_type:nobles_estate
		}
		save_scope_as = origin_country
		ruler_or_regent = {
			save_scope_as = target_character
		}
		c:SER = {
			save_scope_as = target_country
			ruler_or_regent ?= {
				save_scope_as = additional_character_for_portrait
			}
		}
		c:BYZ = {
			save_scope_as = target_country2
			ruler_or_regent ?= {
				save_scope_as = additional_character_for_portrait2
			}
		}
	}

	option = {
		name = flavor_byz_ser.4.a
		
		ai_will_select = {
			value = 100

			if = {
				limit = {
					subject_loyalty <= 50
				}
				value = 0
			}
		}

		add_liberty_desire = liberty_desire_mild_minus

		if = {
			limit = {
				subject_loyalty <= 50
			}
			add_stability = stability_severe_penalty
		}

		c:BYZ ?= {
			trigger_event_non_silently = {
				id = flavor_byz_ser.5
			}
		}

		c:SER ?= {
			trigger_event_non_silently = {
				id = flavor_byz_ser.5
			}
		}
	}

	option = {
		name = flavor_byz_ser.4.b
		
		ai_will_select = {
			value = 100

			if = {
				limit = {
					subject_loyalty > 50
				}
				value = 0
			}
		}

		if = {
			limit = {
				subject_loyalty > 50
			}
			add_stability = stability_severe_penalty
		}

		show_as_tooltip = {
			make_subject_of = {
				target = c:BYZ
				type = subject_type:vassal
			}
		}

		c:BYZ ?= {
			trigger_event_non_silently = {
				id = flavor_byz_ser.6
			}
		}

		c:SER ?= {
			trigger_event_non_silently = {
				id = flavor_byz_ser.6
			}
		}
	}
}

flavor_byz_ser.6 = {
	type = country_event
	title = flavor_byz_ser.6.title
	desc = flavor_byz_ser.6.desc
	outcome = neutral

	illustration_tags = {
		10 = exterior
	}
	
	option = {
		name = flavor_byz_ser.6.byz
		
		trigger = {
			tag = BYZ
		}

		scope:origin_country = {
			make_subject_of = {
				target = ROOT
				type = subject_type:vassal
			}
			if = {
				limit = {
					has_country_modifier = receiving_agitate_for_liberty
				}
				remove_country_modifier = receiving_agitate_for_liberty
			}
		}
		if = {
			limit = {
				has_scripted_relation = {
					target = scope:origin_country
					type = relation_type:agitate_for_liberty
				}
			}
			remove_relation = {
				first = ROOT
				second = scope:origin_country
				type = relation_type:agitate_for_liberty
			}
		}

		add_prestige = prestige_mild_bonus
		add_legitimacy = legitimacy_weak_bonus
	}

	option = {
		name = flavor_byz_ser.6.ser
	
		trigger = {
			tag = SER
		}

		show_as_tooltip = {
			scope:origin_country = {
				make_subject_of = {
					target = scope:target_country2
					type = subject_type:vassal
				}
			}
		}

		add_prestige = prestige_mild_penalty
	}
	trigger = {
		current_date > 476.9.4
	}
}
