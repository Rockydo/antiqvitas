namespace = flavor_hun_ven

scripted_effect hun_ven_list_of_provinces = {
	province_definition:lika_province = {
		every_location_in_province_definition = {
			limit = {
				NOT = {
					this = location:rijeka
				} 
			}
			add_to_list = list_of_dalmatian_locations
		}
	}
	province_definition:north_dalmatia_province = {
		every_location_in_province_definition = {
			add_to_list = list_of_dalmatian_locations
		}
	}
	province_definition:south_dalmatia_province = {
		every_location_in_province_definition = {
			add_to_list = list_of_dalmatian_locations
		}
	}
}

flavor_hun_ven.1 = { #Treaty of Zadar - Hungarian Event
	type = country_event
	title = flavor_hun_ven.1.title
	desc = flavor_hun_ven.1.desc
	outcome = neutral
    historical_info = flavor_hun_ven.1.historical_info

	fire_only_once = yes
	
	dynamic_historical_event = {
		tag = HUN
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 5
	}
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
	}

	illustration_tags = {
		10 = regular
		10 = interior
	}
	
	trigger = {
		current_date > 476.9.4
		at_war = no
		country_exists = c:VEN
		country_exists = c:RAG
		c:CRO ?= {
			OR = {
				in_union_with = c:HUN
				is_subject_of = c:HUN
			}
		}
		c:RAG = {
			is_subject_of = c:VEN
		}
		c:VEN = {
			at_war = no
			OR = {
				stability < 0
				government_power < 50
				war_exhaustion_percentage > 0.25
			}
			owns = location:pag
			owns = location:zadar
			owns = location:split
		}
	}
	
	option = {
		name = flavor_hun_ven.1.a
		historical_option = yes
		add_casus_belli = {
			target = location:pag.owner
			type = casus_belli:cb_conquer_province
			province = location:pag.province
			years = 20
		}
		add_casus_belli = {
			target = location:zadar.owner
			type = casus_belli:cb_conquer_province
			province = location:zadar.province
			years = 20
		}
		add_casus_belli = {
			target = location:split.owner
			type = casus_belli:cb_conquer_province
			province = location:split.province
			years = 20
		}
	}
	
	option = {
		name = flavor_hun_ven.1.b
		c:VEN = {
			trigger_event_non_silently = flavor_hun_ven.2
		}
	}
}

flavor_hun_ven.2 = { #Treaty of Zadar - Venetian Event
	type = country_event
	title = flavor_hun_ven.2.title
	desc = flavor_hun_ven.2.desc
	outcome = neutral
	historical_info = flavor_hun_ven.1.historical_info

	illustration_tags = {
		10 = angry
		10 = interior
	}
	
	major = yes
	major_trigger = { 
		OR = {
			tag = HUN 
			tag = RAG
		}
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		hun_ven_list_of_provinces = yes
	}

	option = {
		name = flavor_hun_ven.2.a
		historical_option = yes
		add_prestige = prestige_mild_bonus
		c:HUN = {
			add_casus_belli = {
				target = location:pag.owner
				type = casus_belli:cb_conquer_province
				province = location:pag.province
				years = 20
			}
			add_casus_belli = {
				target = location:zadar.owner
				type = casus_belli:cb_conquer_province
				province = location:zadar.province
				years = 20
			}
			add_casus_belli = {
				target = location:split.owner
				type = casus_belli:cb_conquer_province
				province = location:split.province
				years = 20
			}
		}
	}
	
	option = {
		name = flavor_hun_ven.2.b
		add_prestige = prestige_severe_penalty
		add_stability = stability_severe_penalty
		c:RAG = {
			make_subject_of = {
				target = c:HUN
				type = subject_type:vassal
			}
		}
		if = {
			limit = {
				c:CRO = {
					OR = {
						in_union_with = c:HUN
						is_subject_of = c:HUN
					}
				}
			}
			every_in_list = {
				list = list_of_dalmatian_locations
				limit = { owner = ROOT }
				change_location_owner = c:CRO
			}
		}
		else = {
			every_in_list = {
				list = list_of_dalmatian_locations
				limit = { owner = ROOT }
				change_location_owner = c:HUN
			}
		}
	}
	trigger = {
		current_date > 476.9.4
	}
}
