namespace = flavor_luc

flavor_luc.1 = { #Silk Manufactoring of Lucca
	type = country_event
	title = flavor_luc.1.title
	desc = flavor_luc.1.desc
	outcome = neutral

	fire_only_once = yes

	dynamic_historical_event = {
		tag = LUC
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 3
	}

	illustration_tags = {
		10 = regular
		10 = exterior
	}

    trigger = {
    	current_date > 476.9.4
		owns = location:lucca
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:burghers_estate }

		ruler_or_regent ?= {
            save_scope_as = target_character
        }
	}

	option = {
		name = flavor_luc.1.a

		location:lucca = {
            construct_building = { #Placeholder building. Should be removed once CAESAR-65011 Jira is implemented
                building_type = building_type:lucca_silk_production_guild
                cost_multiplier = 0.5
                cost_multiplier_reason = "burghers_cheap_sponsorship"
            }
        }

        add_estate_satisfaction = {
            type = estate_type:burghers_estate
            value = estate_satisfaction_mild_bonus
        }
	}

	option = {
		name = flavor_luc.1.b

		add_estate_satisfaction = {
            type = estate_type:burghers_estate
            value = estate_satisfaction_mild_penalty
        }
	}
}

flavor_luc.3 = {
	type = country_event
	title = flavor_luc.3.title
	desc = flavor_luc.3.desc
	outcome = neutral

	dynamic_historical_event = {
		tag = LUC
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 3
	}

	fire_only_once = yes

	trigger = {
		current_date > 476.9.4
		location:london.top_owner = {
			this != root
			NOT = { is_at_war_with = root }
			within_diplomatic_range = root
		}
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:burghers_estate }

		location:london.top_owner = {
			save_scope_as = target_country
		}

		location:london.market = {
			save_scope_as = target_market
		}
	}

	option = {
		name = flavor_luc.3.a
		
		location:london = {
			construct_building = {
				building_type = building_type:trade_office
				cost_multiplier = 0.25
				cost_multiplier_reason = "burghers_cheap_sponsorship"
				owner = root
				payer = root
			}
		}
	}

	option = {
		name = flavor_luc.3.b
		
		add_opinion_mutual_effect = {
			modifier = luc_did_not_intervene_in_trade

		}

		change_societal_value = {
			type = belligerent_vs_conciliatory
			value = societal_value_move_to_right
		}
	}
}

#Propose to be sold to Pisa. Pisa is easier to break free from
flavor_luc.1000 = {
	type = country_event
	title = flavor_luc.1000.title
	desc = flavor_luc.1000.desc
	outcome = neutral

	dynamic_historical_event = {
		tag = LUC
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 20 #This should be really likely to happen
	}

	fire_only_once = yes

	trigger = {
		current_date > 476.9.4
		is_subject = yes
		NOT = {
			is_subject_of = c:PIS
		}

		country_exists = c:PIS

		overlord = {
			at_war = yes
			NOT = { is_at_war_with = c:PIS }
		}
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:burghers_estate }

		c:PIS ?= {
			ruler_or_regent ?= {
				save_scope_as = target_character
			}
		}

		overlord = {
			save_scope_as = target_country
			ruler_or_regent ?= {
				save_scope_as = target_character2
			}
		}

		save_scope_as = from_country
	}

	option = { #Suggest to overlord for the purchase
		name = flavor_luc.1000.a
		
		overlord = {
			trigger_event_non_silently = {
				id = flavor_luc.1001
			}
		}
	}

	option = { #No need to do that. We have better odds with the current overlord
		name = flavor_luc.1000.b
		
		add_prestige = prestige_mild_bonus
	}
}

flavor_luc.1001 = {
	type = country_event
	title = flavor_luc.1001.title
	desc = flavor_luc.1001.desc
	outcome = neutral

	major = yes
	major_trigger = {
		this = c:LUC	
	}

	immediate = {
		
		ordered_known_country = {
			order_by = military_strength
			limit = {
				is_at_war_with = root
			}

			max = 1
			check_range_bounds = yes
			save_scope_as = enemy_country
		}
	}

	option = {
		name = flavor_luc.1001.a
		
		c:PIS ?= {
			trigger_event_non_silently = {
				id = flavor_luc.1002
			}
		}
	}

	option = {
		name = flavor_luc.1001.b
		
		add_opinion_mutual_effect = {
			modifier = luc_refused_to_sell_us
			target = scope:from_country
		}
	}
	trigger = {
		current_date > 476.9.4
	}
}
