namespace = d008_flavor_byz

d008_flavor_byz.1 = { #Possibility to sponsor Greek Fire
	type = country_event
	title = d008_flavor_byz.1.title
	desc = d008_flavor_byz.1.desc
	outcome = neutral
	fire_only_once = yes

	image = "gfx/interface/illustrations/event/backgrounds/special/greek_fire.dds"

	dynamic_historical_event = {
		tag = BYZ
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 5
	}

	trigger = {
		current_date > 476.9.4
		has_or_had_tag = BYZ
		has_dlc = "d008_fate_of_the_phoenix"
		any_artist = {
			artist_type = scientist
			has_art_in_progress = no
			NOT = { has_character_modifier = busy_modifier }
			artist_skill > 0.7
		}
	}

	immediate = {
		random_artist = {
			limit = {
				artist_type = scientist
				has_art_in_progress = no
				NOT = { has_character_modifier = busy_modifier }
				artist_skill > 0.7
			}
			save_scope_as = target_artist
		}
	}

	option = { #Sponsor
		name = d008_flavor_byz.1.a
		change_gold_effect = { scale = -5 }
		custom_tooltip = {
			text = d008_flavor_byz.1.a.tt
			if = {
				limit = { exists = scope:target_artist }
				scope:target_artist = {
					add_character_modifier = {
						modifier = busy_modifier
						years = -1
						mode = add
					}
					set_variable = searching_for_greek_fire
				}
				random_list = {
					50 = {
						trigger_event_non_silently = { id = d008_flavor_byz.2 days = { 60 180 } }
					}
					50 = {
						trigger_event_non_silently = { id = d008_flavor_byz.3 days = { 60 180 } }
					}
				}
			}
		}
		add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_mild_bonus}
		add_estate_satisfaction = { type = estate_type:burghers_estate value = estate_satisfaction_mild_bonus}
	}

	option = { #Nah
		name = d008_flavor_byz.1.b
		add_prestige = prestige_mild_bonus
		add_navy_tradition = navy_tradition_mild_bonus
	}
}

				50 = {
					trigger_event_non_silently = { id = d008_flavor_byz.3 days = { 60 180 } }

d008_flavor_byz.6 = { # A Trebizond-formed Byzantium restores the Empire, earning the Fate of the Phoenix rewards
	type = country_event
	title = d008_flavor_byz.6.title
	desc = d008_flavor_byz.6.desc
	outcome = positive
	fire_only_once = yes

	image = "gfx/interface/illustrations/disaster/fate_of_the_phoenix.dds"

	dynamic_historical_event = {
		tag = BYZ
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 10
	}

	trigger = {
		current_date > 476.9.4
		has_dlc = "d008_fate_of_the_phoenix"
		has_or_had_tag = TRE
		NOT = { has_variable = won_fate_of_the_phoenix_disaster_variable }

		# The success ("phoenix_risen") conditions of fate_of_the_phoenix_end_trigger
		NOT = { is_situation_active = situation:rise_of_the_ottomans }
		num_locations_owned_or_owned_by_subjects_or_below >= 150
		custom_tooltip = {
			text = remember_no_turk_tt
			region:anatolia_region = {
				NOT = {
					any_country_with_capital_in_geography = {
						this != root
						OR = {
							has_reform = government_reform:anatolian_beylik
							culture = culture:turkish_culture
							culture = culture:turkoman_culture
						}
						num_locations_owned_or_owned_by_subjects_or_below > 100
					}
				}
			}
		}
	}

	immediate = {
		ruler_or_regent ?= { save_scope_as = target_character }
	}

	option = {
		name = d008_flavor_byz.6.a

		custom_tooltip = {
			text = tre_restored_empire_unlocks_tt
			set_variable = won_fate_of_the_phoenix_disaster_variable
		}

		custom_tooltip = {
			text = tre_unlock_phoenix_cabinet_actions_tt
			set_variable = unlock_spread_philhellenism_variable
			set_variable = encourage_latin_militarism_variable
		}

		unlock_policy_effect = { type = byz_tagmata_policy_upgraded }

		add_stability = stability_mild_bonus
		add_legitimacy = legitimacy_mild_bonus
	}
}
