flavor_ath.2 = { # Sudden Death of the Duke Guillem II
    type = country_event
    title = flavor_ath.2.title
    desc = flavor_ath.2.desc
    outcome = neutral

    fire_only_once = yes
    dynamic_historical_event = {
        tag = ATH
        from = 476.9.4 
        to = 476.9.4 
        monthly_chance = 60
    }


    trigger = {
    	current_date > 476.9.4
        has_ruler = yes
        is_subject = yes
        has_heir = yes
        country_exists = c:ARG
        country_exists = c:SIC
        overlord = c:SIC
        character:ath_william = {
            is_alive = yes
            is_ruler = yes
        }
        c:ARG = {
            has_ruler = yes
        }
    }

    illustration_tags = {
        10 = regular
        10 = interior
    }	

    immediate = {
        event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
        root = {
            save_scope_as = scope_athens
        }
        ruler ?= {
            save_scope_as = scope_ruler
        }
        heir ?= {
            save_scope_as = scope_heir
        }
        if = {
            limit = {
                any_character = {
                    NOT = { is_child_of = root.ruler }
                    NOT = { is_sibling_of = root.ruler }
                    is_courtier = yes
                    is_female = no
                    is_alive = yes
                    is_adult = yes
                    culture = culture:catalan
                    religion = religion:catholic
                    has_estate = estate_type:nobles_estate
                }
            }
            random_character = {
                limit = {
                    NOT = { is_child_of = root.ruler }
                    NOT = { is_sibling_of = root.ruler }
                    is_courtier = yes
                    is_female = no
                    is_alive = yes
                    is_adult = yes
                    culture = culture:catalan
                    religion = religion:catholic
                    has_estate = estate_type:nobles_estate
                }
                save_scope_as = regent_of_athens
                set_variable = ath_regent_of_athens
            }
        }
        else = {
            create_character = {
                adm = { 50 70 }
                dip = { 40 63 }
                mil = { 50 74 }
                age = { 45 49 }
                culture = culture:catalan
                religion = religion:catholic
                estate = estate_type:nobles_estate
                save_scope_as = regent_of_athens
                set_variable = ath_regent_of_athens
                add_character_modifier = { modifier = block_marriage years = 20 mode = add_and_extend }
            }
        }         
        scope:scope_heir = {
            move_country = c:SIC
        }
        c:SIC ?= {
            save_scope_as = sic_country
        }
        c:ARG ?= {
            save_scope_as = arg_country
            ruler ?= {
                save_scope_as = arg_ruler
            }
            save_scope_as = argos_scope
        }
    }
    

    option = { # Catalan rule must continue with <Sicily heir>
        name = flavor_ath.2.a    
        historical_option = yes  
        kill_character = {
            target = scope:scope_ruler
            reason = disease
        }
        add_stability = stability_mild_bonus
        add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_weak_bonus }      
        ai_chance = {
            factor = 10
            modifier = {
                factor = 0.10
                C:ARG ?= {
                    is_human = yes
                }
            }
        }        
    }

    option = { # Invite the de Brienne to rule
        name = flavor_ath.2.b
        kill_character = {
            target = scope:scope_ruler
            reason = disease
        }
        add_legitimacy = legitimacy_mild_penalty
        ai_chance = {
            factor = 1
            modifier = {
                factor = 10
                C:ARG ?= {
                    is_human = yes
                }
            }
        }        
        custom_tooltip = ath_sicily_may_declare_war
        custom_tooltip = ath_legal_heir_of_guillem
        scope:argos_scope = { trigger_event_non_silently = { id = flavor_ath.3 days = 3 } } 
    }    
}
