flavor_ach.3 = {
    type = country_event
    title = flavor_ach.3.title
    desc = flavor_ach.3.desc
    outcome = neutral

    trigger = {
    	current_date > 476.9.4
        is_latin_state_check = yes
        legitimacy <= 35
        estate_satisfaction:peasants_estate <= 0.25
        ruler ?= { has_dynasty = yes }
        is_subject = no
        NOT = { has_variable = had_greek_insurrection }
    }

    illustration_tags = {
        10 = exterior
        10 = angry
    }

    immediate = {
        event_illustration_estate_effect = {
            foreground = estate_type:peasants_estate
            background = estate_type:peasants_estate
        }
        set_variable = had_greek_insurrection
        ruler = {
            save_scope_as = target_character
        }
        create_character = {
            estate = estate_type:nobles_estate
            culture = culture:greek_culture
            religion = religion:orthodox

            dip = { 40 70 }
            mil = { 60 80 }
            save_scope_as = target_character2
        }
    }

    option = {
        name = flavor_ach.3.a
        
        ai_will_select = {
            value = 40
        }

        set_new_ruler = scope:target_character2
        change_culture = culture:greek_culture
        change_religion = religion:orthodox

            limit = {
                has_estate_privilege = estate_privilege:lat_compromise_with_greek_populace
            }
            revoke_estate_privilege = estate_privilege:lat_compromise_with_greek_populace
         }
        if = {
            limit = {
                has_estate_privilege = estate_privilege:lat_latin_crusader_elite
            }
            revoke_estate_privilege = estate_privilege:lat_latin_crusader_elite
         }
    }

    option = {
        name = flavor_ach.3.b
        
        ai_will_select = {
            value = 60
        }

        change_gold_effect = {
            scale = -6
        }

        add_legitimacy = legitimacy_mild_bonus
    }
}
