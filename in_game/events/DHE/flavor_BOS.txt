namespace = flavor_bos

flavor_bos.1 = { #Aspiration for Independence
	type = country_event
	title = flavor_bos.1.title
	desc = flavor_bos.1.desc
	outcome = neutral

	illustration_tags = {
		10 = armed
		10 = exterior
	}

	fire_only_once = yes
	
	historical_info = flavor_bos.1.historical_info

	dynamic_historical_event = {
		tag = BOS
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 10
	}

	trigger = {
		current_date > 476.9.4
		has_ruler = yes
		country_exists = c:HUN
		is_subject_of = c:HUN
		character:hun_karoly_robert ?= {
			is_alive = no
		}
		c:HUN ?= {
			has_ruler = yes
		}
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		ruler = {
			save_scope_as = target_ruler
		}
		character:hun_karoly_robert ?= {
			save_scope_as = hun_robert
		}
	}

	option = {
		name = flavor_bos.1.a
		historical_option = yes
		custom_tooltip = flavor_bos.1.a.tt
		scope:target_ruler = {
			add_character_modifier = {
				modifier = bos_devils_student
				years = -1
				mode = add
			}
		}
		every_country = {
			limit = {
				OR = {
					is_rival_of = c:HUN
					is_enemy_of = c:HUN
				}
			}
			add_opinion = { target = root modifier = opinion_bos_aspirations_of_liberty }
		}
	}
	
	option = {
		name = flavor_bos.1.b
		add_prestige = prestige_mild_penalty
		add_liberty_desire = liberty_desire_severe_minus
	}
}

flavor_bos.2 = { #The Decline of Serbia
	type = country_event
	title = flavor_bos.2.title
	desc = flavor_bos.2.desc
	outcome = neutral

	illustration_tags = {
		10 = armed
		10 = exterior
	}
	
	historical_info = flavor_bos.2.historical_info

	fire_only_once = yes

	dynamic_historical_event = {
		tag = BOS
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 10
	}

	trigger = {
		current_date > 476.9.4
		has_ruler = yes
		is_independent_or_autonomous_subject = yes
		any_neighbor_country = {
			country_economical_base < c:BOS.country_economical_base
			NOT = { is_allied_with = { target = root } }
			culture = culture:serbian
		}
		OR = {
			AND = {
				country_exists = c:SER
				country_economical_base > c:SER.country_economical_base
				c:SER ?= {
					num_locations < 30
				}
			}
			NOT = {
				country_exists = c:SER
			}
		}
		character:ser_stefan_dusan ?= {
			is_alive = no
		}
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		ruler = {
			save_scope_as = target_ruler
		}
		character:ser_stefan_dusan ?= {
			save_scope_as = ser_dusan
		}
	}

	option = {
		name = flavor_bos.2.a
		historical_option = yes
		change_societal_value = { type = belligerent_vs_conciliatory value = societal_value_move_to_left }
		every_neighbor_country = {
			limit = {
				country_economical_base < c:BOS.country_economical_base
				NOT = { is_allied_with = { target = root } }
				culture = culture:serbian
			}
			root = {
				add_casus_belli = {
					target = prev
					type = casus_belli:cb_annex
				}
			}
		}
	}
	
	option = {
		name = flavor_bos.2.b
		add_stability = stability_mild_bonus
		change_societal_value = {
			type = belligerent_vs_conciliatory
			value = societal_value_move_to_right
		}
	}
}

flavor_bos.3 = { #The First King of Bosnia
	type = country_event
	title = flavor_bos.3.title
	desc = flavor_bos.3.desc
	outcome = neutral

	illustration_tags = {
		10 = armed
		10 = exterior
	}
	
	historical_info = flavor_bos.3.historical_info

	fire_only_once = yes
	
	immediate = {
		ruler = {
			save_scope_as = target_ruler
		}
	}

	dynamic_historical_event = {
		tag = BOS
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 10
	}

	trigger = {
		current_date > 476.9.4
		has_ruler = yes
		ruler = {
			is_female = no
		}
		is_independent_or_autonomous_subject = yes
		country_rank_level <= 2
		modifier:block_from_change_to_kingdom_rank = no
		OR = {
			any_subject_or_below = {
				culture = culture:serbian
			}
			area:serbia_area = {
				any_location_in_area = {
					owner ?= root
					count > 15
				}
			}
		}
	}
	
	option = {
		name = flavor_bos.3.a
		historical_option = yes
		add_legitimacy = legitimacy_severe_bonus
		set_country_rank = country_rank:rank_kingdom
		every_neighbor_country = {
			limit = {
				NOT = {
					is_allied_with = { target = root }
					is_subject_of = root
				}
			}
			add_opinion = {
				target = root
				modifier = opinion_bos_proclaimed_kingdom_of_bosnia
			}
		}
		change_societal_value = { type = belligerent_vs_conciliatory value = societal_value_large_move_to_left }
	}
	
	option = {
		name = flavor_bos.3.b
		change_societal_value = { type = belligerent_vs_conciliatory value = societal_value_large_move_to_right }
		every_neighbor_country = {
			limit = {
				NOT = {
					is_allied_with = { target = root }
					is_subject_of = root
				}
			}
			add_opinion = {
				target = root
				modifier = opinion_bos_diplomatic_restraint
			}
		}
	}
}
