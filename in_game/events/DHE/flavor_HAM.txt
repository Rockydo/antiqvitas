namespace = flavor_ham

flavor_ham.1 = { # The Berenberg Bank
	type = country_event
	title = flavor_ham.1.title
	desc = flavor_ham.1.desc
	outcome = neutral
	historical_info = flavor_ham.1.historical_info
	fire_only_once = yes

	major = yes
	dynamic_historical_event = {
		tag = HAM
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 10
	}

	trigger = {
		current_date > 476.9.4
		owns = location:hamburg
		has_embraced_institution = institution:banking
	}

	illustration_tags = {
        10 = regular
        10 = interior
    }

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:burghers_estate }
		location:hamburg = {
			save_scope_as = berenberg_bank_hq
		}

		create_character = {
			dynasty = dynasty:berenberg_dynasty
			age = { 20 40 }
			save_scope_as = target_character
		}

		scope:berenberg_bank_hq = {
			create_building_country_in_location = {
				name = {
					name = BERENBERG
				}
				reforms = {
					banking_country
				}
				save_scope_as = berenberg_bank_tag
				set_new_ruler = scope:target_character

				hidden_effect = {
					change_country_flag = BERENBERG
					define_unique_country_tag = BERBE
					change_heir_selection = heir_selection:bank_selection
					change_government_type = government_type:republic
				}
			}
			scope:berenberg_bank_hq = {
				change_building_level_in_location = {
					building = building_type:banking_office
					value = 1
					owner = scope:berenberg_bank_tag
				}
			}
		}
		set_local_variable = {
			name = amount_of_gold_for_berenberg
			value = { 250 500 }
		}
	}

	option = {
		name = flavor_ham.1.a

		historical_option = yes

		show_as_tooltip = {
			scope:berenberg_bank_hq = {
				change_building_level_in_location = {
					building = building_type:banking_office
					value = 1
					owner = scope:berenberg_bank_tag
				}
			}
		}
		change_gold_effect = { scale = 2 }
	}

	option = {
		name = flavor_ham.1.b
		high_risk_option = yes

		show_as_tooltip = {
			scope:berenberg_bank_hq = {
				change_building_level_in_location = {
					building = building_type:banking_office
					value = 1
					owner = scope:berenberg_bank_tag
				}
			}
		}
		change_gold_effect = { scale = 2 }

		change_player = scope:berenberg_bank_tag
	}

	after = {
		scope:berenberg_bank_tag = {
			trigger_event_non_silently = flavor_ham.2
		}
	}
}
