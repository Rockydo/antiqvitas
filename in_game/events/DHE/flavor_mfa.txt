# Theodore's Passing
flavor_mfa.1 = {
	type = country_event
	title = flavor_mfa.1.title
	desc = flavor_mfa.1.desc
	outcome = neutral

	fire_only_once = yes
	dynamic_historical_event = {
		tag = MFA
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 100
	}

	major = yes
	major_trigger = {
		capital ?= {
			area = area:piemonte_area
		}
	}

	trigger = {
		current_date > 476.9.4
		character:mfa_teodoro_i_palaiologos = {
			is_alive = no
		}
		exists = c:BYZ
		has_ruler = yes
		ruler.dynasty ?= dynasty:palaiologos_dynasty
	}

	immediate = {
		ruler = {
			save_scope_as = target_character
			dynasty = {
				save_scope_as = target_dynasty
			}
		}
		character:mfa_teodoro_i_palaiologos = {
			save_scope_as = target_character2
		}
		c:BYZ = {
			save_scope_as = target_country
		}
		government_reform:byzantine_court_in_italy = {
			save_scope_as = target_government_reform
		}

		scope:target_country = {
			event_illustration_estate_foreground_effect = {
				foreground = estate_type:nobles_estate
			}
			event_illustration_estate_background_effect = {
				background = estate_type:nobles_estate
			}
		}

		capital = {
			random_pop = {
				limit = {
					pop_type = pop_type:nobles
				}
				save_scope_as = target_pop_for_conversion
			}
		}
	}

	illustration_tags = {
		10 = interior
		10 = happy
	}

	option = {
		name = flavor_mfa.1.a

		ai_will_select  = {
			value = 95
		}
		
		historical_option = yes
		
		culture = {
			add_cultural_tradition = cultural_influence_extreme_bonus
		}
		add_government_power = government_power_mild_bonus
	}

	option = {
		name = flavor_mfa.1.b
		
		ai_will_select  = {
			value = 5
		}

		unlock_government_reform_effect = {
			type = byzantine_court_in_italy
		}

		scope:target_character = {
			change_character_culture = culture:greek_culture
			change_character_religion = religion:orthodox

			every_descendant = {
				change_character_culture = culture:greek_culture
				change_character_religion = religion:orthodox
			}
			every_spouse = {
				change_character_culture = culture:greek_culture
				change_character_religion = religion:orthodox
			}
		}
		
		change_culture = culture:greek_culture
		change_religion = religion:orthodox

		add_reform = scope:target_government_reform

		custom_tooltip = mfa_extra_information_about_reform

		custom_tooltip = mfa_opportunity_for_thessaloniki

		if = {
			limit = {
				exists = scope:target_pop_for_conversion
			}
			scope:target_pop_for_conversion = {
				split_pop = {
					fraction = 0.1
					culture = culture:greek_culture
					religion = religion:orthodox
				}
			}
		}

		add_opinion = {
			modifier = we_are_kin
			target = scope:target_country
		}
	}
}

	add_prestige = prestige_mild_bonus

		mode = add_and_extend

	}

		}

flavor_mfa.3 = {
	type = country_event
	title = flavor_mfa.3.title
	desc = flavor_mfa.3.desc
	outcome = neutral

	trigger = {
		current_date > 476.9.4
		ruler ?= {
			this != character:mfa_teodoro_i_palaiologos
		}
		cultural_unity >= 0.5
	}

	fire_only_once = yes
	dynamic_historical_event = {
		tag = MFA
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 10
	}

	major = yes
	major_trigger = {
		capital ?= {
			area = area:piemonte_area
		}
	}

	immediate = {
		save_scope_as = from_country
		ruler = {
			save_scope_as = target_character
		}
		province_definition:pavia_province = {
			every_location_in_province_definition = {
				add_to_list = affected_locations
			}
		}
		province_definition:novara_province = {
			every_location_in_province_definition = {
				add_to_list = affected_locations
			}
		}
		province_definition:monferrato_province = {
			every_location_in_province_definition = {
				add_to_list = affected_locations
			}
		}
		location:alessandria = {
			add_to_list = affected_locations
		}
		location:tortona = {
			add_to_list = affected_locations
		}
		location:novi = {
			add_to_list = affected_locations
		}
		location:alba = {
			add_to_list = affected_locations
		}
		location:mondovi = {
			add_to_list = affected_locations
		}
		location:cuneo = {
			add_to_list = affected_locations
		}
		location:ivrea = {
			add_to_list = affected_locations
		}
		every_in_list = {
			list = affected_locations
			owner ?= {
				add_to_list = affected_countries
			}
		}
		event_illustration_estate_effect = {
			background = estate_type:nobles_estate
			foreground = estate_type:nobles_estate
		}
	}

	illustration_tags = {
		10 = interior
		10 = regular
	}

	option = {
		name = flavor_mfa.3.a

		ai_will_select  = {
			value = 30
		}
		
		drop_antagonism_bomb = {
			modifier = antagonism_embraced_antagonism
			target = capital
			value = 10
		}

		add_diplomats = diplomats_mild_penalty

		every_in_list = {
			list = affected_countries
			limit = {
				NOR = {
					this = ROOT
					any_overlord_or_above = {
						this = ROOT
					}
				}
			}
			ordered_owned_location = {
				limit = {
					is_in_list = affected_locations
				}
				order_by = population
				max = 1
				check_range_bounds = no
				ROOT = {
					add_casus_belli = {
						target = prev.owner
						type = casus_belli:cb_conquer_province
						province = prev.province
						years = 20
					}
				}
			}
		}

		custom_tooltip = {
			text = mfa_if_we_conquer_enough_in_20_years
			trigger_event_non_silently = {
				id = flavor_mfa.4
				years = 20
			}
			trigger_event_non_silently = {
				id = flavor_mfa.6
				years = 20
			}
		}
	}

	option = {
		name = flavor_mfa.3.b

		ai_will_select  = {
			value = 70
		}
	
		change_societal_value = {
			type = belligerent_vs_conciliatory
			value = societal_value_large_move_to_right
		}
	}
}

					limit = {
						owner ?= scope:target_country
					}
					order_by = population
					max = 1
					check_range_bounds = no
					save_scope_as = location_for_release
				}
			}
		}
		else_if = {
			limit = {
				scope:variable_holder = { has_variable = mfa_provinces }
			}
			scope:target_country = {
				ordered_owned_location = {
					order_by = population
					limit = {
						OR = {
							province_definition = province_definition:chalkidiki_province
							province_definition = province_definition:lower_macedonia_province
							province_definition = province_definition:central_macedonia_province
						}
					}
					max = 1
					check_range_bounds = no
					save_scope_as = location_for_release
				}
			}
		}

			error_log = "Reached unreachable script in flavor_mfa.1005"
		}
	}

	illustration_tags = {
		10 = regular
		10 = interior
	}

		name = flavor_mfa.1005.a

			value = 75
		}

			limit = {
