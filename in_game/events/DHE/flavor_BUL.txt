flavor_bul.1 = { #Principality of Dobruja
	type = country_event
	title = flavor_bul.1.title
	desc = flavor_bul.1.desc
	outcome = neutral

	fire_only_once = yes

	historical_info = flavor_bul.1.historical_info

	dynamic_historical_event = {
		tag = BUL
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 10
	}

	trigger = {
		current_date > 476.9.4
		any_active_disaster = { disaster_type = disaster_type:twilight_of_the_tsardom }
		NOT = { country_exists = c:KVN }
	}
	
	illustration_tags = {
		10 = angry
		10 = exterior
	}
	
	immediate = {
		ruler = {
			save_scope_as = target_ruler
		}
		if = {
			limit = {
				NOT = {
					dynasty_exists = terter_dynasty
				}
			}
			location:kaliakra = { create_dynasty_from_location = terter_dynasty }
		}
		create_character = {
			first_name = Balik
			min_age = 40
			birth_location = location:kaliakra
			estate = estate_type:nobles_estate
			dynasty = dynasty:terter_dynasty
			adm = { 20 60 }
			dip = { 20 60 }
			mil = { 50 80 }
			culture = culture:bulgarian
			save_scope_as = target_character
			script = bul_balik
			create_in_limbo = yes
		}
		every_owned_location = {
			limit = {
				OR = {
					province = location:constanta.province
					this = location:pangalia
					this = location:karvuna
					this = location:kaliakra
					this = location:varna
				}
			}
			add_core = c:KVN
		}
	}

	option = {
		name = flavor_bul.1.a
		historical_option = yes
		add_stability = stability_mild_bonus
		add_legitimacy = legitimacy_severe_penalty
		add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_mild_bonus }
		change_societal_value = { type = centralization_vs_decentralization value = societal_value_move_to_right }
        create_country_from_cores_in_our_locations = c:KVN
		c:KVN = {
			set_new_ruler = scope:target_character
			make_subject_of = {
				target = root
				type = subject_type:vassal
			}
		}
	}	
	
	option = {
		name = flavor_bul.1.b
		add_stability = stability_severe_penalty
		add_legitimacy = legitimacy_mild_bonus
		add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_severe_penalty }
		change_societal_value = { type = centralization_vs_decentralization value = societal_value_move_to_left }
		create_rebel = {
			name = dobrujan_revolt
			category = estate
			estate = nobles_estate
			save_scope_as = target_rebel
		}
		scope:target_character = { change_character_allegiance = scope:target_rebel }
		custom_tooltip = {
			text = flavor_bul.1.b.tt
			every_owned_location = {
				limit = {
					is_core_of = c:KVN
				}
				every_pop = {
					limit = {
						owner = root
					}
					change_pop_allegiance = scope:target_rebel
					add_pop_satisfaction = pop_satisfaction_severe_penalty
				}
			}
		}
	}
	
	option = {
		name = flavor_bul.1.c
		high_risk_option = yes
		add_stability = stability_mild_bonus
		add_legitimacy = legitimacy_severe_penalty
		add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_mild_bonus }
		change_societal_value = { type = centralization_vs_decentralization value = societal_value_move_to_right }
        create_country_from_cores_in_our_locations = c:KVN
		c:KVN = {
			set_new_ruler = scope:target_character
			make_subject_of = {
				target = root
				type = subject_type:vassal
			}
		}
		change_player = c:KVN
	}	
}
