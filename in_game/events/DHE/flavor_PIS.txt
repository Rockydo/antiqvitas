flavor_pis.4 = { #Completing the Leaning Tower of Pisa
	type = country_event
	title = flavor_pis.4.title
	desc = flavor_pis.4.desc

	dynamic_historical_event = {
		tag = PIS
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 3
	}

	fire_only_once = yes

	trigger = {
		current_date > 476.9.4
		work_of_art:leaning_tower_of_pisa ?= {
			owner ?= ROOT
		}
	}

	outcome = neutral

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:burghers_estate }

		if = {
			limit = {
				any_artist = {
					artist_type = architect
				}
			}

			random_artist = {
				limit = {
					artist_type = architect
				}
				save_scope_as = target_character
			}
		}
		else = {
			create_character = {
				estate = estate_type:burghers_estate
				age = { 25 35 }
				artist_skill = { 0.5 0.9 }
				artist = architect
				save_scope_as = target_character
			}
		}
	}

	option = {
		name = flavor_pis.4.a
		
		add_prestige = prestige_extreme_bonus

		work_of_art:leaning_tower_of_pisa = {
			change_art_quality = 15
		}

		change_gold_effect = {
			scale = -2
		}
	}

	option = {
		name = flavor_pis.4.b
		
		add_prestige = prestige_mild_penalty
	}
}

#A Shift Towards Land Focus
flavor_pis.5 = {
	type = country_event
	title = flavor_pis.5.title
	desc = flavor_pis.5.desc

	dynamic_historical_event = {
		tag = PIS
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 3
	}

	fire_only_once = yes

	trigger = {
		current_date > 476.9.4
		owns = location:pisa
		
		total_effective_building_levels:porto_pisano <= 1

		any_neighbor_country = {
			NOT = { is_subject_of = root }
			NOT = { has_casus_belli_on = this }
			OR = {
				country_type = army
				country_type = location
			}
		}
	}

	outcome = neutral

	immediate = {
		ruler_or_regent ?= {
			save_scope_as = target_character
		}
	}

	option = {
		name = flavor_pis.5.a

		custom_tooltip = {
			text = pis_gain_casus_belli_on_every_neighbor_country_in_italy
			every_neighbor_country = {
				limit = {
					NOT = { is_subject_of = root }
					NOT = { has_casus_belli_on = this }
					OR = {
						country_type = army
						country_type = location
					}
				}
				root = {
					add_casus_belli = {
						target = prev
						type = casus_belli:cb_conquer_province
						province = prev.capital.province
					}
				}
			}
		}
		
		change_societal_value = {
			type = belligerent_vs_conciliatory
			value = societal_value_move_to_left
		}

		change_societal_value = {
			type = land_vs_naval
			value = societal_value_move_to_left
		}

		add_country_modifier = {
			modifier = pis_condottieri_contracts
			years = 5
		}
	}

	option = {
		name = flavor_pis.5.b
		
		capital ?= {
			change_development = development_mild_bonus
			change_prosperity = prosperity_mild_bonus
		}

		add_stability = stability_mild_bonus

		change_societal_value = {
			type = belligerent_vs_conciliatory
			value = societal_value_move_to_right
		}

		change_societal_value = {
			type = land_vs_naval
			value = societal_value_move_to_right
		}
	}
}

#Gioco del Ponte
#A repeatable event about the Gioco del Ponte festival in Pisa
flavor_pis.6 = {
	type = country_event
	title = flavor_pis.6.title
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {	
					has_variable = gioco_del_ponte_promising_leader
				}
				desc = flavor_pis.6.desc.a
			}
			triggered_desc = {
				trigger = {	
					has_variable = gioco_del_ponte_satisfied_population
				}
				desc = flavor_pis.6.desc.b
			}
			triggered_desc = {
				trigger = {	
					has_variable = gioco_del_ponte_spread_of_culture
				}
				desc = flavor_pis.6.desc.c
			}
		}
	}

	trigger = {
		current_date > 476.9.4
		current_year >= 1550	

		has_or_had_tag = PIS

		owns = location:pisa
	}

	outcome = positive

	immediate = {
		random_list = {
			1 = {
				set_variable = gioco_del_ponte_promising_leader
				create_character = {
					age = { 25 40 }
					adm = { 30 50 }
					dip = { 30 50 }
					mil = { 50 70 }
					create_in_limbo = yes
					save_scope_as = target_character
				}
			}
			1 = {
				set_variable = gioco_del_ponte_satisfied_population
			}
			1 = {
				set_variable = gioco_del_ponte_spread_of_culture
			}
		}
	}

	option = {
		name = flavor_pis.6.a

		trigger = {
			has_variable = gioco_del_ponte_satisfied_population
		}

		add_stability = stability_mild_bonus
		
		location:pisa = {
			every_pop = {
				limit = {
					owner ?= root	
				}
				add_pop_satisfaction = pop_satisfaction_extreme_bonus
			}
		}
	}

	option = {
		name = flavor_pis.6.b
		
		trigger = {
			has_variable = gioco_del_ponte_spread_of_culture
		}

		add_stability = stability_mild_bonus

		culture = {
			add_cultural_influence = cultural_influence_mild_bonus
			add_cultural_tradition = cultural_tradition_mild_bonus
		}

		add_prestige = prestige_mild_bonus
	}

	option = {
		name = flavor_pis.6.c

		trigger = {
			has_variable = gioco_del_ponte_promising_leader
		}
		
		add_stability = stability_mild_bonus

		scope:target_character = {
			move_country = root
		}
	}

	option = {
		name = flavor_pis.6.d

		trigger = {
			has_variable = gioco_del_ponte_promising_leader
		}
		
		add_stability = stability_extreme_bonus

		kill_character_silently = {
			target = scope:target_character
		}
	}

	after = {
		remove_variable = gioco_del_ponte_promising_leader
		remove_variable = gioco_del_ponte_satisfied_population
		remove_variable = gioco_del_ponte_spread_of_culture
	}
}
