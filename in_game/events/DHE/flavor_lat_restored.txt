flavor_lat.1 = { # The New Latin Bureaucracy
    type = country_event
    title = flavor_lat.1.title
    desc = flavor_lat.1.desc
    outcome = neutral

    fire_only_once = yes
    dynamic_historical_event = {
        tag = LAT
        from = 476.9.4

     option = { # For a new distinct Latin Bureaucracy

            base = 5

            base = 5
        } 
        add_religious_influence = religious_influence_mild_bonus

    } 

    option = { # Model the new bureaucracy after the Byzantines

        ai_chance = {
            base = 5
        } 
        add_legitimacy = legitimacy_mild_bonus
        add_country_modifier = { modifier = lat_byzantine_inspired_bureaucracy years = 25 mode = add }

    option = { # Implement a Catalan inspired bureaucracy

flavor_lat.2 = { # The Rising Turkish Threat in Anatolia
    type = country_event
    title = flavor_lat.2.title
    desc = flavor_lat.2.desc
    outcome = neutral
    image = "gfx/interface/illustrations/situation/rise_of_the_ottomans.dds"

    fire_only_once = yes
    dynamic_historical_event = {
        tag = LAT
        from = 476.9.4 
        to = 476.9.4 
        monthly_chance = 40
    }


    trigger = {
    	current_date > 476.9.4
        at_war = no
        has_ruler = yes
        owns = location:constantinople
        any_known_country = {
            is_neighbor_of = ROOT
            AND = {
                culture = culture:turkish_culture
                num_locations > 20
                OR = {
                    has_ruler = yes
                    has_regent = yes
                }
            }
        }
    }

    immediate = {
        ruler ?= {
            save_scope_as = scope_ruler
        }
        location:constantinople = {
            save_scope_as = constantinople_location
        } 
        random_known_country = {
            limit = {
                is_neighbor_of = ROOT
                AND = {
                    culture = culture:turkish_culture
                    num_locations > 20 
                    OR = {
                        has_ruler = yes
                        has_regent = yes
                    }
                }
            }
            save_scope_as = rising_turkish_country
            ruler_or_regent ?= {
                save_scope_as = rising_turkish_country_ruler
            }
        }  
    }
    

    option = { # We shall accomplish what our predecessors and drive back the Beyliks!
        name = flavor_lat.2.a         
        change_societal_value = { type = offensive_vs_defensive value = societal_value_move_to_left }
        add_country_modifier = { modifier = lat_strike_the_beyliks years = 15 mode = add }   
        add_casus_belli = {
            target = scope:rising_turkish_country
            type = casus_belli:cb_annex
        }   
    }

    option = { # We should remain wary, and take a more defensive approach
        name = flavor_lat.2.b         
        change_societal_value = { type = offensive_vs_defensive value = societal_value_move_to_right }
        add_country_modifier = { modifier = lat_defensive_approach_to_anatolia years = 15 mode = add }      
    } 
}
