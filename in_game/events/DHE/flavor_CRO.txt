namespace = flavor_cro

flavor_cro.1000 = { #The Unification of Croatia and Slavonia - Croatia
	type = country_event
	title = flavor_cro.1000.title
	desc = flavor_cro.1000.desc
	outcome = neutral
    historical_info = flavor_cro.1000.historical_info

	fire_only_once = yes
	
	dynamic_historical_event = {
		tag = CRO
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 5
	}
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
	}

	illustration_tags = {
		10 = regular
		10 = interior
	}
	
	trigger = {
		current_date > 476.9.4
		at_war = no
		country_exists = c:TUR
		country_exists = c:HUN
		OR = {
			in_union_with = c:HUN
			is_subject_of = c:HUN
		}
		any_neighbor_country = {
			tag = TUR
		}
		c:HUN = {
			any_owned_location = {
				area = area:slavonia_area
				is_core_of = c:SVN
			}
		}
	}
	
	option = {
		name = flavor_cro.1000.a
		historical_option = yes
		c:HUN = {
			trigger_event_non_silently = flavor_cro.1001
		}
	}
	
	option = {
		name = flavor_cro.1000.b
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_left
		}
	}
}

flavor_cro.1001 = { #The Unification of Croatia and Slavonia - Croatia
	type = country_event
	title = flavor_cro.1001.title
	desc = flavor_cro.1001.desc
	outcome = neutral
	historical_info = flavor_cro.1000.historical_info

	illustration_tags = {
		10 = angry
		10 = interior
	}
	
	major = yes
	major_trigger = { 
		tag = CRO
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		region:balkan_region = {
			every_location_in_region = {
				limit = {
					has_owner = yes
					OR = {
						area = area:slavonia_area
						is_core_of = c:SVN
					}
				}
				add_to_list = list_of_slavonian_locations
			}
		}
	}

	option = {
		name = flavor_cro.1001.a
		historical_option = yes
		custom_tooltip = {
			text = flavor_cro.1001.a.tt
			every_in_list = {
				list = list_of_slavonian_locations
				limit = { owner ?= ROOT }
				change_location_owner = c:CRO
			}
		}
		c:CRO = {
			add_opinion = {
				target = root
				modifier = cro_approved_the_merge
			}
			change_societal_value = {
				type = offensive_vs_defensive
				value = societal_value_move_to_right
			}
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_left
		}
	}
	
	option = {
		name = flavor_cro.1001.b
		c:CRO = {
			add_opinion = {
				target = root
				modifier = cro_rejected_the_merge
			}
		}
		change_societal_value = {
			type = centralization_vs_decentralization
			value = societal_value_move_to_right
		}
	}
	trigger = {
		current_date > 476.9.4
	}
}

flavor_cro.1100 = { #The Dominion of Ivan Nelipic
	type = country_event
	title = flavor_cro.1100.title
	desc = flavor_cro.1100.desc
	outcome = neutral
    historical_info = flavor_cro.1100.historical_info

	fire_only_once = yes
	
	dynamic_historical_event = {
		tag = CRO
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 10
	}
	
	illustration_tags = {
		10 = happy
		10 = exterior
	}
	
	trigger = {
		current_date > 476.9.4
		at_war = no
		country_exists = c:HUN
		in_union_with = c:HUN
	}
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		
		if = {
			limit = {
				NOT = {
					dynasty_exists = nelipic_dynasty
				}
			}
			location:knin = { create_dynasty_from_location = nelipic_dynasty }
		}
		
		create_character = {
			first_name = name_john
			religion = religion:catholic
			estate = estate_type:nobles_estate
			min_age = 40
			female = no
			birth_location = location:knin
			culture = culture:croatian
			dynasty = dynasty:nelipic_dynasty
			adm = 85
			dip = 73
			mil = 79
			save_scope_as = target_character
			create_in_limbo = yes
		}
	}
	
	option = {
		name = flavor_cro.1100.a
		historical_option = yes
		add_legitimacy = legitimacy_mild_penalty
		kill_character_silently = scope:target_character
	}
	
	option = {
		name = flavor_cro.1100.b
		add_legitimacy = legitimacy_mild_bonus
		c:HUN = {
			trigger_event_non_silently = flavor_cro.1101
		}
	}
}
