namespace = mercs
mercs.1 = {
	hide_portraits = no
	type = country_event
	title = mercs.1.title
	desc = mercs.1.desc
	outcome = neutral
	# image = city in ruins
	trigger = {
		current_date > 476.9.4
		any_hired_mercenary ?= {
			count > 0
			mercenary_home ?= {
				has_owner = yes
				NOT = { owner = root }
				owner = { is_at_war_with = root }
			}
		}
	}


	immediate = {
		hidden_effect = {
		
			random_hired_mercenary = {
				limit = {
					mercenary_home = {
						NOT = { owner = root }
						owner = { is_at_war_with = root }
					}
				}
				save_scope_as = patriotic_mercs
				mercenary_home = {
					save_scope_as = merc_home_province
					owner = {
						save_scope_as = merc_home_country
					}
				}
			}
		}
	}

	option = {
		name = mercs.1.a
		scope:patriotic_mercs = {
			every_mercenary_sub_unit = {
				add_subunit_strength_percentage = -0.2
			}
		}
	}
	option = {
		name = mercs.1.b
		trigger = {
			scope:patriotic_mercs = {
				OR = {
					culture = culture:highland
					culture = culture:scottish
				}
			}
		}
		add_gold = {
			value = scope:patriotic_mercs.monthly_cost
			multiply = -10
		}
	}
}

mercs.2 = {
	type = country_event
	title = mercs.2.title
	desc = mercs.2.desc
	outcome = neutral
	    illustration_tags = {
        10 = armed
        10 = exterior
    }

	trigger = {
		current_date > 476.9.4
		NOT = { has_variable = bloodthirsty_mercs_recent }
		any_hired_mercenary ?= {
			count > 0
			mercenary_home ?= {
				has_owner = yes
				NOT = { owner = root }
				owner = {
					any_rival = {
						is_at_war_with = root
					}
				}
			}
		}
	}


	immediate = {
		set_variable = { name = bloodthirsty_mercs_recent value = 1 years = 2 }
		hidden_effect = {
			random_hired_mercenary = {
				limit = {
					mercenary_home = {
						NOT = { owner = root }
						owner = {
							any_rival = {
								is_at_war_with = root
							}
						}
					}
				}


				save_scope_as = bloodthirsty_mercs

				mercenary_home = {
					save_scope_as = bloodthirsty_mercs_home_province
					owner = {
						save_scope_as = bloodthirsty_mercs_home
						random_rival = {
							limit = {
								is_at_war_with = root
							}
							save_scope_as = bloodthirsty_mercs_enemy
						}
					}
				}
			}
		}event_illustration_estate_effect = {
    foreground = estate_type:burghers_estate
    background = estate_type:peasants_estate
	#background = estate_type:army_estate
	}
	}

	option = {
		name = mercs.2.a
		scope:bloodthirsty_mercs = {
			every_mercenary_sub_unit = {
				add_subunit_strength_percentage = 0.10
			}
		}
		add_prestige = prestige_weak_bonus
	}

}

mercs.3 = {
	type = country_event
	title = mercs.3.title
	desc = mercs.3.desc
	outcome = neutral
	    illustration_tags = {
        10 = armed
        10 = interior
    }
	
	trigger = {
		current_date > 476.9.4
		at_war = yes
		NOT = { has_variable = mercs_allowed_to_loot_recent }
		any_hired_mercenary ?= {
			count > 0
			mercenary_home ?= {	NOT = { owner ?= root }	}
		}
		any_country = {
			is_at_war_with = root
			any_owned_non_rural_location = {
				development >= 20
			}
		}
	}


	immediate = {
		set_variable = { name = mercs_allowed_to_loot_recent value = 1 years = 2 }
		hidden_effect = {
			random_hired_mercenary = {
				limit = {
					mercenary_home = {	NOT = { owner = root }	}
				}
				save_scope_as = loot_mercs
			}


			random_country = {
				limit = {
					is_at_war_with = root
					any_owned_location = {
						development >= 20
					}
				}
				random_owned_location = {
					limit = {
						development >= 20
					}
					save_scope_as = rich_enemy_province
				}
			}
		}event_illustration_estate_effect = {
    foreground = estate_type:burghers_estate
    background = estate_type:burghers_estate
	#background = estate_type:army_estate
	}}

	option = {
		name = mercs.3.a
		add_gold = {
			value = scope:rich_enemy_province.development
			multiply = 2
		}
		add_stability = stability_mild_penalty
		scope:loot_mercs = {
			every_mercenary_sub_unit = {
				add_subunit_strength_percentage = -0.05
			}
		}
	}
	option = {
		name = mercs.3.b
		add_gold = {
			value = scope:loot_mercs.monthly_cost
			multiply = -10
		}
	}
}

mercs.4 = {
	type = country_event
	title = mercs.4.title
	desc = mercs.4.desc
	outcome = neutral
		    illustration_tags = {
        10 = happy
        10 = interior
    }
	
	trigger = {
		current_date > 476.9.4
		at_war = yes
		any_hired_mercenary ?= {
			count > 0
			any_mercenary_sub_unit = {
				subunit_strength_percentage < 0.7
			}
		}
		
		prestige >= 40.00
	}


	immediate = {
		hidden_effect = {
			random_hired_mercenary = {
				limit = {
					any_mercenary_sub_unit = {
						subunit_strength_percentage < 0.7
					}
				}
				save_scope_as = veteran_mercs
			}
		}event_illustration_government_estate_effect = yes
	}

	option = {
		name = mercs.4.a
		scope:veteran_mercs = {
			ordered_mercenary_sub_unit = {
				order_by = {
					value = subunit_strength_percentage
					multiply = -1
				}
				max = 10
				check_range_bounds = no
					
				add_subunit_strength_percentage = 0.2
			}
		}
	}
}

mercs.10 = {
	type = country_event
	title = mercs.10.title
	desc = mercs.10.desc
	outcome = neutral
	illustration_tags = {
        10 = armed
        10 = interior
    }
	
	trigger = {
		current_date > 476.9.4

		at_war = yes
		NOT = { has_variable = pious_mercenaries_recent }

		any_hired_mercenary ?= {
			count > 0
			mercenary_home ?= {
				NOT = { owner = root }
				dominant_religion = root.religion
			}
		}

		any_country_at_war_with = {
			NOT = { religion = root.religion }
		}
	}


	immediate = {
		set_variable = { name = pious_mercenaries_recent value = 1 years = 2 }
		random_hired_mercenary = {
			limit = {
				mercenary_home = {
					NOT = { owner = root }
					dominant_religion = root.religion
				}
			}
			save_scope_as = pious_mercs
		}

		random_country_at_war_with = {
			limit = {
				NOT = { religion = root.religion }
			}
			save_scope_as = pious_mercs_enemy
		}
		event_illustration_estate_effect = {
    foreground = estate_type:clergy_estate
    background = estate_type:clergy_estate
	}}

	option = {
		name = mercs.10.a
		scope:pious_mercs = {
			every_mercenary_sub_unit = {
				add_subunit_strength_percentage = 0.15
			}
		}
		add_army_tradition = army_tradition_weak_bonus
		add_prestige = prestige_weak_bonus
	}
}

#mercs suffering a lot of attrition, thinking of abandoning their contracts
mercs.11 = {
	type = country_event
	title = mercs.11.title
	desc = mercs.11.desc
	outcome = neutral
	orphan = yes
	illustration_tags = {
        10 = armed
        10 = interior
    }
	
	option = { #let them go
		name = mercs.11.a
		every_in_list = {
			list = sub_units
			destroy_subunit = yes
		}
		add_country_modifier = {
			modifier = mercenaries_abandoned
			years = 10
			mode = replace
		}
	}
	option = { #get emergency supplies
		name = mercs.11.b
		custom_tooltip = {
			text = mercs_bailed_out
			save_temporary_scope_value_as = {
				name = total_cost
				value = 0
			}
			every_in_list = {
				list = sub_units
				save_temporary_scope_value_as = {
					name = total_cost
					value = {
						value = scope:total_cost
						add = {
							value = mercenary_maintenance_cost
						}
					}
				}
				mercenary = {
					rescue_sub_unit = prev
				}
			}
			save_temporary_scope_value_as = {
				name = total_cost
				value = {
					value = scope:total_cost
					multiply = 3 #3 months' worth to bail them out
				}
			}
		}
		add_gold = {
			value = scope:total_cost
			multiply = -1
		}
		add_country_modifier = {
			modifier = mercenaries_rescued
			years = 10
			mode = replace
		}
	}
	trigger = {
		current_date > 476.9.4
	}
}
