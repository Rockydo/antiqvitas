latinization_vs_hellenization.1 = {	#Hellenic Literature spreads
	type = country_event
	title = latinization_vs_hellenization.1.title
	desc = latinization_vs_hellenization.1.desc
	outcome = neutral
	trigger = {
		current_date > 476.9.4
		has_societal_value = societal_value_type:latinization_vs_hellenization
		"culture_percentage_in_country(culture:greek_culture)" > 0
		has_dlc = "d008_fate_of_the_phoenix"
	}

	illustration_tags = {
		10 = angry
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }

		ordered_owned_location = {
			limit = {
				culture_percentage = {
					culture = culture:greek_culture
					value > 0
				}
			}
			order_by = {
				value = {
					add = population
					divide = {
						value = rank_index
						min = 1
					}
				}
			}
			check_range_bounds = no
			max = root.num_provinces
			add_to_list = revival_location_list
		}

		random_in_list = {
			list = revival_location_list

			save_scope_as = target_location
			province = {
				every_location_in_province = {
					ordered_pop = {
						limit = { 
							culture != culture:greek_culture 
							owner = root
						}
						order_by = {
							value = pop_size
							if = {
								limit = { is_upper_class = yes }
								multiply = 1000
							}
						}
						check_range_bounds = no
						max = 4
						add_to_list = non_greek_pops_list
					}
				}
			}
		}

		create_character = {
			culture = root.culture
			birth_location = scope:target_location
			age = 30
			adm = { 25 100 }
			dip = { 25 100 }
			mil = { 25 100 }

			create_in_limbo = yes
			save_scope_as = target_character
		}
	}

	option = {	#Encourage these publications
		name = latinization_vs_hellenization.1.a

		change_societal_value = {
			type = traditionalist_vs_innovative
			value = societal_value_minor_move_to_right
		}

		if = {
			limit = {
				list_size = {
					name = non_greek_pops_list
					value > 0
				}
			}
			every_in_list = {
				list = non_greek_pops_list
				split_pop = {
					fraction = { 0.2 0.5 }
					culture = culture:greek_culture
				}
			}
		}

		add_research_progress = research_progress_mild_bonus

		scope:target_character = {
			move_country = root
		}
	}

	option = {	#Best to not spread these cultural ideas
		name = latinization_vs_hellenization.1.b

		if = {
			limit = { culture != culture:greek_culture }
			culture = {
				add_cultural_influence = cultural_influence_mild_bonus
				add_cultural_tradition = cultural_tradition_mild_bonus
			}
		}
		

		custom_tooltip = {
			text = target_character_disappears_tt
			kill_character_silently = scope:target_character
		}
	}
}

		societal_value:latinization_vs_hellenization > -100

			has_country_modifier = latinization_currents_modifier

		10 = regular

	}

	option = {
		name = latinization_vs_hellenization.2.a

		add_country_modifier = {
			modifier = latinization_currents_modifier
			years = modifier_duration_years_extended
			mode = replace
		}

		if = {
			limit = {
				OR = {
					societal_value:latinization_vs_hellenization > 25
					societal_value:latinization_vs_hellenization < -25
				}
			}				
			add_stability = {
				value = -1

					value = societal_value:latinization_vs_hellenization
					multiply = 0.1

			custom_tooltip = stability_scales_with_current_latinization_hellenization
		}
	}

	option = {	#Better to not shift our values
		name = latinization_vs_hellenization.2.b

		add_country_modifier = {
			modifier = suppressing_latinization_modifier
			years = modifier_duration_years_normal
			mode = extend
		}

		custom_tooltip = while_this_modifier_active_reduced_latinization_gain_from_events_tt
	}
}

latinization_vs_hellenization.3 = {	#Latinization Pays Off
	type = country_event
	title = latinization_vs_hellenization.3.title
	desc = latinization_vs_hellenization.3.desc
	outcome = neutral
	trigger = {
		current_date > 476.9.4
		has_societal_value = societal_value_type:latinization_vs_hellenization
		modifier:monthly_towards_latinization > modifier:monthly_towards_hellenization
		societal_value:latinization_vs_hellenization < 0
		has_dlc = "d008_fate_of_the_phoenix"
	}

	illustration_tags = {
		10 = happy
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:burghers_estate }

		random_list = {
			1 = { 
				trigger = { stability < 99 }
				set_variable = societal_values_3_1
			}
			1 = { 
				set_variable = societal_values_3_2 
				ordered_character = {
					limit = { 
						total_abilities < 300 
						can_serve_in_cabinet_of = root
					}
					order_by = {
						value = {
							add = total_abilities
							if = {
								limit = { in_cabinet = yes }
								multiply = 2
							}
							subtract = age_in_years
						}
					}
					check_range_bounds = no
					max = 1
					save_scope_as = target_character
				}
			}
			1 = { set_variable = societal_values_3_3 }
			1 = { set_variable = societal_values_3_4 }
		}
	}

	option = {
		name = latinization_vs_hellenization.3.a


		switch = {
			trigger = has_variable
			societal_values_3_1 = {
				if = {
					limit = { has_country_modifier = suppressing_latinization_modifier }
					add_stability = stability_mild_bonus

					custom_tooltip = bonus_weakened_because_of_suppressing_latinization_modifier
				}
				else = { add_stability = stability_severe_bonus } 
			}
			societal_values_3_2 = {
				if = {
					limit = { has_country_modifier = suppressing_latinization_modifier }
					scope:target_character = {
						
						add_adm = { 2 5 }
						add_dip = { 2 5 }
						add_mil = { 2 5 }
					}
					custom_tooltip = bonus_weakened_because_of_suppressing_latinization_modifier
				}
				else = {
					scope:target_character = {
						add_adm = { 3 10 }
						add_dip = { 3 10 }
						add_mil = { 3 10 }
					}
				} 
			}
			societal_values_3_3 = {
				if = {
					limit = { has_country_modifier = suppressing_latinization_modifier }
					culture = {
						add_cultural_influence = cultural_influence_weak_bonus
						add_cultural_tradition = cultural_tradition_weak_bonus
					}

					custom_tooltip = bonus_weakened_because_of_suppressing_latinization_modifier
				}
				else = {
					culture = {
						add_cultural_influence = cultural_influence_mild_bonus
						add_cultural_tradition = cultural_tradition_mild_bonus
					}
				} 
			}
			societal_values_3_4 = {
				if = {
					limit = { has_country_modifier = suppressing_latinization_modifier }
					add_prestige = prestige_weak_bonus

					custom_tooltip = bonus_weakened_because_of_suppressing_latinization_modifier
				}
				else = { add_prestige = prestige_mild_bonus } 
			}
		}
	}

	option = {
		name = latinization_vs_hellenization.3.b

		trigger = { has_country_modifier = suppressing_latinization_modifier }

		remove_country_modifier = suppressing_latinization_modifier
	}

	after = {
		hidden_effect = {
			remove_variable = societal_values_3_1
			remove_variable = societal_values_3_2
			remove_variable = societal_values_3_3
			remove_variable = societal_values_3_4
		}
	}
}

		if = {
			limit = { has_variable = LvH_event_28_latin_trend_variable }
			#Current drift favours Latinization → nobles lean Hellenic
			change_societal_value = {
				type = latinization_vs_hellenization
				value = societal_value_minor_move_to_right
			}
		}
		else_if = {
			limit = { has_variable = LvH_event_28_greek_trend_variable }
			#Current drift favours Hellenization → nobles lean Latin
			change_societal_value = {
				type = latinization_vs_hellenization
				value = societal_value_minor_move_to_left
			}
		}
	}

		add_estate_satisfaction = {
			type = estate_type:nobles_estate
			value = estate_satisfaction_mild_penalty

			limit = { has_variable = LvH_event_28_latin_trend_variable }
			#Reinforce Latin drift
			change_societal_value = {
				type = latinization_vs_hellenization
				value = societal_value_tiny_move_to_left
			}
		}
		else_if = {
			limit = { has_variable = LvH_event_28_greek_trend_variable }
			#Reinforce Greek drift
			change_societal_value = {
				type = latinization_vs_hellenization
				value = societal_value_tiny_move_to_right
			}
		}
	}

latinization_vs_hellenization.30 = {    #Peasants Resist Current Trend
    type = country_event
    title = latinization_vs_hellenization.30.title
    desc = latinization_vs_hellenization.30.desc
    outcome = neutral
    trigger = {
    	current_date > 476.9.4
        has_societal_value = societal_value_type:latinization_vs_hellenization
        country_has_estate = estate_type:peasants_estate
		has_dlc = "d008_fate_of_the_phoenix"
        OR = {
            societal_value:latinization_vs_hellenization > 15
            societal_value:latinization_vs_hellenization < -15
        }
    }

    illustration_tags = {
        10 = angry
        10 = exterior
    }

    immediate = {
        event_illustration_estate_effect = { 
            foreground = estate_type:peasants_estate 
            background = estate_type:peasants_estate 
        }

        # Choose 1–3 high-pop rural provinces where peasants exist
		ordered_pop = {
			limit = { 
				owner = root
				is_upper_class = no
				location = { proximity > 0 }
			}
			order_by = pop_size
			max = {
                value = {
                    add = root.num_provinces
                    multiply = 0.1
                    ceiling = yes
                    min = 1
                    max = 6
                }
            }
            check_range_bounds = no
            add_to_list = LvH_event_30_pops_list
		}

		ordered_in_list = {	#Saving these scopes for reference in the description of the event :)
			list = LvH_event_30_pops_list

			order_by = pop_size
			max = 1
			check_range_bounds = no

			save_scope_as = target_pop

			location = { save_scope_as = target_location }
		}
    }

    option = {   #Appease the peasant communities
        name = latinization_vs_hellenization.30.a

        every_in_list = {
			list = LvH_event_30_pops_list

			add_pop_satisfaction = pop_satisfaction_mild_bonus
		}

        add_estate_satisfaction = { 
            type = estate_type:peasants_estate 
            value = estate_satisfaction_mild_bonus
        }

        # Ideological impact depends on what trend the low class opposes
        if = {
            limit = { societal_value:latinization_vs_hellenization > 0 }

            # Peasants resist Latinization
            change_societal_value = {
                type = latinization_vs_hellenization
                value = societal_value_move_to_left
            }
        }
        else = {
            # Peasants resist Hellenization
            change_societal_value = {
                type = latinization_vs_hellenization
                value = societal_value_move_to_right
            }
        }
    }

    option = {   #Force compliance with the cultural trend
        name = latinization_vs_hellenization.30.b

        every_in_list = {
			list = LvH_event_30_pops_list

			add_pop_satisfaction = pop_satisfaction_mild_penalty
		}

        add_estate_satisfaction = { 
            type = estate_type:peasants_estate 
            value = estate_satisfaction_severe_penalty
        }

        # Repression reinforces the current dominant trend
        if = {
            limit = { societal_value:latinization_vs_hellenization > 0 }

            change_societal_value = {
                type = latinization_vs_hellenization
                value = societal_value_tiny_move_to_right
            }
        }
        else = {
            change_societal_value = {
                type = latinization_vs_hellenization
                value = societal_value_tiny_move_to_left
            }
        }

        add_stability = stability_weak_penalty
    }
}
