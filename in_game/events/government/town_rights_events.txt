town_rights_events.1 = {	# Staple Port of [target_location.GetNameWithNoTooltip] Grows
	type = country_event
	title = town_rights_events.1.title
	desc = town_rights_events.1.desc
	outcome = positive

	trigger = {
		current_date > 476.9.4
		any_town_rights_in_country = {
			this.town_rights_type = town_rights_type:staple_port
			location = { NOT = { has_variable = had_good_staple_port_event_variable } }
		}

		at_war = no
		is_during_bankruptcy = no
	}

	illustration_tags = {
		10 = happy
		10 = exterior
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:burghers_estate }

		random_locations_with_town_rights_in_country = {
			limit = { 
				has_town_rights = town_rights_type:staple_port 
				NOT = { has_variable = had_good_staple_port_event_variable }
			}
			save_scope_as = target_location

			set_variable = {
				name = had_good_staple_port_event_variable
				years = 5
			}

			province = { save_scope_as = target_province }
		}

		ruler_or_regent = { save_scope_as = target_character }
	}

	option = {
		name = town_rights_events.1.a

		ai_will_select = { value = 1000 }

		trigger = {
			ruler_or_regent = {
				OR = {
					has_trait = lawgiver
					adm > 85
				}
			}
		}
		scope:target_province = {
			every_location_in_province = {

				ordered_pop = {
					limit = { owner = root pop_type = pop_type:burghers }
					order_by = pop_size
					max = 1
					check_range_bounds = no

					add_pop_satisfaction = pop_satisfaction_weak_bonus
				}
			}
		}

		custom_tooltip = {
			text = every_location_in_province_dev_and_prosperity_tt
			scope:target_province = {
				every_location_in_province = {
					change_development = development_weak_bonus
					change_prosperity = prosperity_mild_bonus
				}
			}
		}

	}

	option = {
		name = town_rights_events.1.b

		ai_will_select = { value = 1 }

		scope:target_location = { change_development = development_mild_bonus }
	}

	option = {
		name = town_rights_events.1.c

		ai_will_select = { value = 1 }

		scope:target_location = { change_prosperity = prosperity_extreme_bonus }
	}
}
