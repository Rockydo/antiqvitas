italian_renaissance_culture.1 = { # To Rival Florentine Wool
    type = country_event
    title = italian_renaissance_culture.1.title
    desc = italian_renaissance_culture.1.desc
    outcome = neutral

    trigger = {
    	current_date > 476.9.4
        culture = { has_culture_group = culture_group:italian_group }
        OR = {
            current_age = age_2_renaissance
            current_age = age_3_discovery
        }
        NOT = { owns = location:florence }
        NOT = { tag = FLO }
        NOT = { tag = TUS }
        NOT = { has_variable = italian_rival_wool_industry }
        region:italy_region = {
            any_location_in_region = {
                AND = {
                    raw_material ?= goods:wool
                    owner ?= ROOT
                }
                count > 4
            }
        }
        country_type = location
    }

    illustration_tags = {
        10 = regular
        10 = interior
    }

    immediate = {
        event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
        set_variable = {
            name = italian_rival_wool_industry
            value = 1
            years = 200
        }  
    }	


    option = { # Invest in our Wool industry
        name = italian_renaissance_culture.1.a
        add_prestige = prestige_mild_bonus
        custom_tooltip = ita_culture_rival_wool_guild_locations.tt
        add_gold = {
            value = root.monthly_income_trade_and_tax
            multiply = -3
        }
        add_gold_to_estate = {
            estate_type = estate_type:burghers_estate
            value = {
                value = root.estate_tax_base:peasants_estate
                multiply = 3
            }
        }
        hidden_effect = {
            every_owned_location = {
                limit = {
                    raw_material = goods:wool
                    region = region:italy_region
                }
                if = {
                    limit = { is_full_expanded_rgo = no }
                    change_max_raw_material_workers = 1
                }
                add_location_modifier = {
                    modifier = italian_culture_rival_wool_trade
                    years = 60
                    mode = add_and_extend
                }
            }
        }
    }

    option = { # Rely on the Estates instead
        name = italian_renaissance_culture.1.b
        add_government_power = government_power_mild_bonus
        add_estate_satisfaction = { type = estate_type:peasants_estate value = estate_satisfaction_mild_bonus }
        add_gold_to_estate = {
            estate_type = estate_type:peasants_estate
            value = {
                value = root.estate_tax_base:peasants_estate
                multiply = 1
            }
        } 
    }
}

italian_renaissance_culture.4 = { # Capital Becomes a Center of Humanism
    type = country_event
    title = italian_renaissance_culture.4.title
    desc = italian_renaissance_culture.4.desc
    outcome = positive

    trigger = {
    	current_date > 476.9.4
        culture = { has_culture_group = culture_group:italian_group }
        OR = {
            current_age = age_2_renaissance
            current_age = age_3_discovery
        }
        societal_value:spiritualist_vs_humanist > 20
        capital = {
            region = region:italy_region
            OR = {
                location_rank = location_rank:town
                location_rank = location_rank:city
                location_rank = location_rank:megalopolis
            }
        }	
        NOT = { has_variable = got_center_of_humanism_event }
        country_type = location
    }

    illustration_tags = {
        10 = regular
        10 = interior
    }

    immediate = {
        event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
        ruler_or_regent ?= {
            save_scope_as = target_character
        }
        capital = {
            save_scope_as = target_location
        }
        set_variable = {
            name = got_center_of_humanism_event
            value = 1
            years = 150
        }  
    }	


    option = { # The Capital shall become a beacon of humanism in Italy
        name = italian_renaissance_culture.4.a
        change_societal_value = { type = spiritualist_vs_humanist value = societal_value_move_to_right }
        change_societal_value = { type = traditionalist_vs_innovative value = societal_value_move_to_right }
        scope:target_location = {
            add_location_modifier = {
                modifier = italian_culture_center_of_humanism
                years = 100
                mode = add_and_extend
            }
            set_variable = {
                name = is_a_center_of_renaissance_humanism
                value = 1
                years = 200
            } 
        }
    }

    option = { # Discourge the spread of humanism
        name = italian_renaissance_culture.4.b
        add_stability = stability_mild_bonus
        change_societal_value = { type = spiritualist_vs_humanist value = societal_value_move_to_left }
        change_societal_value = { type = traditionalist_vs_innovative value = societal_value_move_to_left }
        add_estate_satisfaction = { type = estate_type:clergy_estate value = estate_satisfaction_mild_bonus }
    }

                mercenary_home = {
                    NOT = { owner = root }
