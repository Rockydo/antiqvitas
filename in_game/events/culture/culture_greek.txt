namespace = culture_greek

scripted_trigger is_part_of_constantinople_patriarchy = {
    is_member_of_international_organization_of_type = {
        type = autocephalous_patriarchate
    }

    autocephalous_patriarchate = {
        var:seat ?= location:constantinople #Check if the Patriarchy if the one of Constantinople, for flavor sake
        has_variable = patriarch
    }
}

scripted_effect save_constantinople_patriarch_as_scope = { #Note: Should never go without the above one, as the former checks if it exists, and this one just saves the scopes
    random_international_organizations_member_of ?= {
        limit = {
            international_organization_type = international_organization_type:autocephalous_patriarchate 
            var:seat ?= location:constantinople
        }
        
        var:patriarch ?= {
            save_scope_as = target_character
        }
    }
}

			months = 10

			owner = root
			to_location = scope:target_destination
		}

		scope:target_character ?= {
			move_country = scope:target_country
		}
	}
