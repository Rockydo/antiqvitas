twilight_of_the_tsardom.5 = { #Local Elites Question our Authority
	category = disaster_event
    type = country_event
	title = twilight_of_the_tsardom.5.title
	desc = twilight_of_the_tsardom.5.desc
	outcome = neutral
	image = "gfx/interface/illustrations/disaster/twilight_of_the_tsardom.dds"
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:burghers_estate background = estate_type:nobles_estate }

			limit = { 
				is_capital = no 
				NOT = {
					has_location_modifier = bul_powerful_local_elites
				}
			}
			save_scope_as = target_location
		}
	}
	
	trigger = {
		current_date > 476.9.4
		any_owned_location = {
			is_capital = no 
			NOT = {
				has_location_modifier = bul_powerful_local_elites
			}
		}
	}

	option = {
		name = twilight_of_the_tsardom.5.a
		add_stability = stability_weak_penalty
		scope:target_location = {
			add_location_modifier = { modifier = bul_powerful_local_elites years = 20 mode = add }
		}
	}

	option = {
		name = twilight_of_the_tsardom.5.b
		scope:target_location = {
			every_pop = {
				limit = {
					owner = root
					pop_type = pop_type:nobles
				}
				add_pop_satisfaction = pop_satisfaction_extreme_penalty
			}
		}
		add_stability = stability_weak_bonus
	}
}

twilight_of_the_tsardom.10 = { #Feuding Estates
	category = disaster_event
	type = country_event
	title = twilight_of_the_tsardom.10.title
	desc = twilight_of_the_tsardom.10.desc
	outcome = neutral
	image = "gfx/interface/illustrations/disaster/twilight_of_the_tsardom.dds"

	trigger = {
		current_date > 476.9.4
		any_estate = {
			NOT = { estate_type = estate_type:crown_estate }
			count >= 2
		}
	}

	illustration_tags = {
        10 = angry
        10 = interior
    }

	immediate = {
		random_estate = {
			limit = {
				NOT = { estate_type = estate_type:crown_estate }
			}
			save_scope_as = first_estate
		}
		random_estate = {
			limit = {
				NOT = { estate_type = estate_type:crown_estate }
				this != scope:first_estate
			}
			save_scope_as = second_estate
		}
	}
	
	option = {
		name = twilight_of_the_tsardom.10.a
		add_estate_satisfaction = { type = scope:first_estate.estate_type value = estate_satisfaction_mild_bonus }
		add_estate_satisfaction = { type = scope:second_estate.estate_type value = estate_satisfaction_mild_penalty }
	}
	
	option = {
		name = twilight_of_the_tsardom.10.b
		add_estate_satisfaction = { type = scope:first_estate.estate_type value = estate_satisfaction_mild_penalty }
		add_estate_satisfaction = { type = scope:second_estate.estate_type value = estate_satisfaction_mild_bonus }
	}
	
	option = {
		name = twilight_of_the_tsardom.10.c
		add_prestige = prestige_mild_bonus
		add_estate_satisfaction = { type = scope:first_estate.estate_type value = estate_satisfaction_severe_penalty }
		add_estate_satisfaction = { type = scope:second_estate.estate_type value = estate_satisfaction_severe_penalty }
	}
}

twilight_of_the_tsardom.11 = { #Surrounded by Vultures
	category = disaster_event
    type = country_event
	title = twilight_of_the_tsardom.11.title
	desc = twilight_of_the_tsardom.11.desc
	outcome = neutral
	image = "gfx/interface/illustrations/disaster/twilight_of_the_tsardom.dds"

	illustration_tags = {
		10 = armed
		10 = exterior
	}

	trigger = {
		current_date > 476.9.4
		any_neighbor_country = {
			NOT = { is_allied_with = { target = ROOT } }
			NOT = { has_truce_with = ROOT }
			any_border_location = {
				any_neighbor_location = {
					owner ?= ROOT
				}
			}
		}
	}
	 
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
		random_neighbor_country = {
			limit = {
				NOT = { is_allied_with = { target = ROOT } }
				NOT = { has_truce_with = ROOT }
				NOT = { root = { has_blocked_treaties = prev } }
			}
			save_scope_as = target_country
		}
		random_border_location = {
			limit = {
				is_neighbor_of = scope:target_country
			}
			province = {
				save_scope_as = border_claim_province
			}
		}
	}

	option = {
		name = twilight_of_the_tsardom.11.a
		scope:target_country = {
			add_opinion = {
				target = root
				modifier = border_friction_from_event
			}
		}
		scope:target_country = {
			add_casus_belli =  {
				type = casus_belli:cb_conquer_province
				province = scope:border_claim_province
				target = ROOT
			}
		}
	}
	
	option = {
		name = twilight_of_the_tsardom.11.b
		add_prestige = prestige_mild_penalty
		transfer_yearly_gold = {
			value = 0.25
			target = scope:target_country
		}
	}
}
