namespace = war_of_the_roses
war_of_the_roses.1 = {
	type = country_event
	category = disaster_event
	title = war_of_the_roses.1.title
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					ruler.dynasty ?= dynasty:lancaster_dynasty
				}
				desc = war_of_the_roses.1.desc.lancaster_ruler
			}
			triggered_desc = {
				trigger = { always = yes }
				desc = war_of_the_roses.1.desc
			}
		}
	}
	outcome = neutral
	image = "gfx/interface/illustrations/disaster/war_of_the_roses.dds"

	immediate = {
		dynasty:york_dynasty = {
			dynasty_head = { save_scope_as = target_york_head }
		}

		dynasty:lancaster_dynasty = {
			dynasty_head = { save_scope_as = target_lancaster_head }
		}
		
		add_estate_satisfaction = { 
			type = estate_type:peasants_estate 
			value = {
				add = estate_satisfaction:peasants_estate
				multiply = -0.75
				max = estate_satisfaction_mild_penalty
				min = estate_satisfaction_unique_penalty
			}
		}

	}

	option = {
		name = war_of_the_roses.1.a
		add_stability = stability_ultimate_penalty

		create_rebel = {
			category = pretender
			name = york_rebel_name
			destroy = no

			save_temporary_scope_as = york_rebel
		}
		
		if = {
			limit = {
				dynasty:lancaster_dynasty = {
					exists = dynasty_head
				}
			}
			set_as_designated_heir_with_reason_effect = {
				target = "dynasty:lancaster_dynasty.dynasty_head"
				reason = preferred_heir_of_ruler
			}
		}

		set_variable = { name = supports_house_of_lancaster value = 1 }

		dynasty:york_dynasty = {
			dynasty_head ?= {
				change_character_allegiance = scope:york_rebel
			}
		}


		hidden_effect = {
			dynasty:york_dynasty = {
				every_character_in_dynasty = {
					limit = {
						this != prev.dynasty_head
					}
					change_character_allegiance = scope:york_rebel
				}
			}

		}


		custom_tooltip = {
			text = york_rebels_joining
			province_definition:north_riding_province = {
				every_location_in_province_definition = {
					limit = { owner ?= root }
					every_pop = {
						limit = { owner = root }
						add_pop_satisfaction = pop_satisfaction_ultimate_penalty
						change_pop_allegiance = scope:york_rebel
					}
				}
			}
			province_definition:west_riding_province = {
				every_location_in_province_definition = {
					limit = { owner ?= root }
					every_pop = {
						limit = { owner = root }
						add_pop_satisfaction = pop_satisfaction_ultimate_penalty
						change_pop_allegiance = scope:york_rebel
					}
				}
			}
		}

		scope:york_rebel = {
			add_rebel_progress = rebel_progress_weak_bonus
		}

	}
	option = {
		name = war_of_the_roses.1.b
		add_stability = stability_ultimate_penalty

		create_rebel = {
			category = pretender
			name = lancaster_rebel_name
			destroy = no
	
			save_temporary_scope_as = lancaster_rebel
		}
		
		if = {
			limit = {
				dynasty:york_dynasty = {
					exists = dynasty_head
				}
			}
			set_as_designated_heir_with_reason_effect = {
				target = "dynasty:york_dynasty.dynasty_head"
				reason = preferred_heir_of_ruler
			}
		}
		
		set_variable = { name = supports_house_of_york value = 1 }
		
		dynasty:lancaster_dynasty = {
			dynasty_head ?= {
				change_character_allegiance = scope:lancaster_rebel
			}
		}

		
		hidden_effect = {
			dynasty:lancaster_dynasty = {
				every_character_in_dynasty = {
					limit = {
						this != prev.dynasty_head
					}
					change_character_allegiance = scope:lancaster_rebel
				}
			}
		}



		custom_tooltip = {
			text = lancaster_rebels_joining
			province_definition:lancashire_province = {
				every_location_in_province_definition = {
					limit = { owner ?= root }
					every_pop = {
						limit = { owner = root }
						add_pop_satisfaction = pop_satisfaction_ultimate_penalty
						change_pop_allegiance = scope:lancaster_rebel
					}
				}
			}
			province_definition:cheshire_province = {
				every_location_in_province_definition = {
					limit = { owner ?= root }
					every_pop = {
						limit = { owner = root }
						add_pop_satisfaction = pop_satisfaction_ultimate_penalty
						change_pop_allegiance = scope:lancaster_rebel
					}
				}
			}
		}

		scope:lancaster_rebel = {
			add_rebel_progress = rebel_progress_weak_bonus
		}

	}

	after = { custom_tooltip = ensuing_chaos_distabilizes_peasants_estate_tt }
	trigger = {
		current_date > 476.9.4
	}
}

scripted_trigger valid_dynasty_alternative = {
	this != dynasty:york_dynasty
	this != dynasty:lancaster_dynasty
	this != dynasty:plantagenet_dynasty
	dynasty_head ?= {
		is_adult = yes
		is_female = no
		NOT = { exists = rebel }
		NOT = { has_variable = wor_kingmaker }
	}
}

war_of_the_roses.6 = {
	type = country_event
	category = disaster_event
	title = war_of_the_roses.6.title
	desc = war_of_the_roses.6.desc
	outcome = neutral
	image = "gfx/interface/illustrations/disaster/war_of_the_roses.dds"
	trigger = {
		current_date > 476.9.4
		any_dynasty = {
			valid_dynasty_alternative = yes
		}
	}
	
	immediate = {
		random_dynasty = {
			limit = {
				valid_dynasty_alternative = yes
			}
			dynasty_head = {
				set_variable = { name = wor_kingmaker value = 1}
				save_scope_as = kingmaker
			}
		}
		
	}

	option = {
		name = war_of_the_roses.6.a
		if = {
			limit = {
				has_variable = supports_house_of_lancaster
			}
			scope:kingmaker = {	
				change_character_allegiance = scope:york_rebel
			}
		}
		if = {
			limit = {
				has_variable = supports_house_of_york
			}
			scope:kingmaker = {	
				change_character_allegiance = scope:lancaster_rebel
			}
		}
	}
	option = {
		name = war_of_the_roses.6.b
		trigger = {
			num_loans <= 0
			legitimacy > 40
			
		}
		add_gold = { 
			value = root.monthly_income_trade_and_tax
			multiply = {
				value = 110
				subtract = legitimacy
				divide = -10
			}
		}
		
		if = {
			limit = {
				has_variable = supports_house_of_lancaster
			}
			scope:york_rebel = {	
				add_rebel_progress = -0.25
			}
		}
		if = {
			limit = {
				has_variable = supports_house_of_york
			}
			scope:lancaster_rebel = {	
				add_rebel_progress = -0.25
			}
		}
		
	}
	
}
