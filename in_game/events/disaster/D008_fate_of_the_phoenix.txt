fate_of_the_phoenix.2 = {	#Smyrna Reclaimed!
	type = country_event
	category = disaster_event
	title = fate_of_the_phoenix.2.title
	desc = fate_of_the_phoenix.2.desc
	outcome = neutral
	historical_info = rise_of_the_ottomans.206.historical_info

	fire_only_once = yes

	image = "gfx/interface/illustrations/disaster/fate_of_the_phoenix.dds"

	trigger = {
		current_date > 476.9.4
		has_dlc = "d008_fate_of_the_phoenix"
	}

	immediate = {
		location:smyrna = { save_scope_as = target_location }

		set_variable = {
			name = target_building_level_variable
			value = {
				add = {
					desc = "BUILDING_LEVEL_BASE"
					value = 1
				}
				add = {
					desc = "BUILDING_LEVEL_DEVELOPMENT"
					value = scope:target_location.development
					multiply = 0.1
				}
				add = {
					desc = "BUILDING_LEVEL_POPULATION"
					value = scope:target_location.population
					multiply = 0.05
				}
				if = {
					limit = {
						scope:target_location.location_rank ?= location_rank:city
					}
					add = {
						desc = "BUILDING_LEVEL_IS_CITY"
						value = 5
					}
				}
				if = {
					limit = {
						scope:target_location.location_rank ?= location_rank:megalopolis
					}
					add = {
						desc = "BUILDING_LEVEL_IS_MEGAPOLIS"
						value = 10
					}
				}
				
				if = {
					limit = { 
						scope:target_location.owner ?= { modifier:market_building_levels != 0 }
					}
					multiply = {
						desc = "BUILDING_LEVEL_FROM_OWNER_MODIFIERS"
						value = 1
						add = scope:target_location.owner.modifier:market_building_levels
					}
				}

				floor = yes
			}
		}
	}

	option = {	#Build the port up
		name = fate_of_the_phoenix.2.a

		scope:target_location = {
			construct_building = {
				building_type = building_type:shipyard
				cost_multiplier = 0
				cost_multiplier_reason = "royal_sponsorship"
			}

			add_location_modifier = {
				modifier = port_of_smyrna_modifier
				years = modifier_duration_years_very_long
				mode = replace
			}
		}
	}

	option = {	#Build the markets up
		name = fate_of_the_phoenix.2.b

		scope:target_location = {
			if = {
				limit = { rank_index < 4 }
				construct_building = {
					building_type = building_type:merchants_quarters
					cost_multiplier = 0
					instant = yes
					cost_multiplier_reason = "royal_sponsorship"
				}
				random_buildings_in_location = {
					limit = { building_type = building_type:merchants_quarters }
					prev = {
						change_building_level_in_location = {
							building = building_type:merchants_quarters
							value = root.var:target_building_level_variable
							owner = root
						}
					}
				}
				custom_tooltip = building_set_to_max_level_tt
			}
			else = { change_development = development_extreme_bonus }

			add_location_modifier = {
				modifier = markets_of_smyrna_modifier
				years = modifier_duration_years_long
				mode = replace
			}
		}
	}

	after = {
		remove_variable = target_building_level_variable
	}
}

	desc = fate_of_the_phoenix.6.desc
	outcome = neutral
	image = "gfx/interface/illustrations/disaster/fate_of_the_phoenix.dds"

	trigger = {
		current_date > 476.9.4
		has_dlc = "d008_fate_of_the_phoenix"
		any_province = {
			NOT = { has_variable = fate_of_the_phoenix_6 }
			NOT = { 
				any_location_in_province = { is_capital = yes }
			}
		}
	}
	immediate = {
		ordered_province = {
			limit = {
				NOT = { has_variable = fate_of_the_phoenix_6 }
				NOT = { 
					any_location_in_province = { is_capital = yes }
				}
			}
			order_by = {
				value = province_average_control
				multiply = -1
			}
			max = 10
			check_range_bounds = no

			add_to_list = fate_of_the_phoenix_6_list
		}

		random_in_list = {
			list = fate_of_the_phoenix_6_list

			save_scope_as = target_province

			set_variable = {
				name = fate_of_the_phoenix_6
				years = 5
			}
		}

		if = {
			limit = { 
				any_cabinet_character = { count > 0 } 
			}
			random_cabinet_character = { save_scope_as = target_character }
		}
	}

	option = {
		name = fate_of_the_phoenix.6.a

		custom_tooltip = {
			text = pops_in_target_province_lose_10_satisfaction_tt
			scope:target_province = {
				every_location_in_province = {
					every_pop = {
						limit = { owner = root }
						add_pop_satisfaction = pop_satisfaction_mild_penalty

				}
			}
		}
	}

	option = {
		name = fate_of_the_phoenix.6.b

			}
		}
	}

	option = {
