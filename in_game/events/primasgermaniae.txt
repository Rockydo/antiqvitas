namespace = primasgermaniae
primasgermaniae.1 = {
	type = country_event
	title = primasgermaniae.1.title
	desc = primasgermaniae.1.desc
	outcome = neutral
	fire_only_once = yes
	
	dynamic_historical_event = {
		tag = MAI
		from = 476.9.4
		to = 476.9.4
		monthly_chance = 1
	}
	
	illustration_tags = {
		10 = interior
		10 = happy
	}

	trigger = {
		current_date > 476.9.4
		is_elector = yes
		religion = religion:catholic
		government_type = government_type:theocracy
		has_ruler = yes
		international_organization:hre = {
			international_organization_has_policy = policy:golden_bull_policy
		}
		country_exists = c:PAP
		c:PAP = {
			opinion = {
				target = root
				value > 150
			}
		}
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }

		save_scope_as = target_country
		ruler ?= { save_scope_as = home_ruler }
	}

	option = {
		name = primasgermaniae.1.a
		add_prestige = prestige_mild_bonus
		international_organization:hre = {
			international_organization_add_special_status = {
				type = special_status:primas_germaniae
				country = root
			}
		}
		add_reform = government_reform:primas_germaniae
	}
}

primasgermaniae.2 = {
	hide_portraits = yes
	type = country_event
	title = primasgermaniae.2.title
	desc = primasgermaniae.2.desc
	outcome = neutral
	
	illustration_tags = {
		10 = interior
		10 = angry
	}
	
	trigger = {
		current_date > 476.9.4
		has_reform = government_reform:primas_germaniae
		country_exists = c:PAP
		prestige < 10
		c:PAP = {
			opinion = { target = root value < -50 }
		}
		has_ruler = yes 
		international_organization:hre = {
			any_international_organization_member = {
				this != root
				government_type = government_type:theocracy
				save_temporary_scope_as = new_target
				c:PAP = {
					"opinion(scope:new_target)" > 150
				}
			}
		}
	}
	
	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
		save_scope_as = target_country
		ruler ?= { save_scope_as = home_ruler }
		international_organization:hre = {
			random_international_organization_member = {
				limit = {
					this != root
					government_type = government_type:theocracy
					save_temporary_scope_as = new_target
					c:PAP = {
						"opinion(scope:new_target)" > 150
					}
				}
				save_scope_as = other_elector
			}
		}
	}
	
	option = {
		name = primasgermaniae.2.a
		save_scope_as = original_prime
		add_prestige = prestige_mild_penalty
		
		scope:other_elector = {
			trigger_event_non_silently = {
				id = primasgermaniae.3
			}
		}
	}
}

primasgermaniae.6 = {
	hide_portraits = yes
	type = country_event
	title = primasgermaniae.6.title
	desc = primasgermaniae.6.desc
	outcome = neutral

	illustration_tags = {
		10 = interior
		10 = happy
	}

	trigger = {
		current_date > 476.9.4
		government_type = government_type:theocracy
		religion = religion:catholic
		
		NOT = {
			international_organization:hre ?= {
				any_country = {
					has_reform = government_reform:primas_germaniae
				}
			}
		}
		country_exists = c:PAP
		c:PAP = { 
			opinion = { target = prev value > 150 }
			has_ruler = yes
		}
		prestige > 10
		has_ruler = yes
	}

	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:clergy_estate background = estate_type:clergy_estate }
		ruler ?= { save_scope_as = home_ruler }
		c:PAP ?= { 
			save_scope_as = pope_country
			ruler ?= { save_scope_as = pope_ruler }
		}
	}

	option = {
		name = primasgermaniae.6.a
		add_prestige = prestige_mild_bonus
		add_reform = government_reform:primas_germaniae
		international_organization:hre = {
			international_organization_add_special_status = {
				type = special_status:primas_germaniae
				country = root
			}
		}
	}
	option = {
		name = primasgermaniae.6.b
		reverse_add_opinion = { target = c:PAP modifier = opinion_declined_primas_germaniae }
	}
}
