colonial_charter.1 = {
	hide_portraits = yes
	type = country_event	

	title = colonial_charter.1.title
	desc = colonial_charter.1.desc
	outcome = neutral
	
	
	
	
	trigger = {
		current_date > 476.9.4
		scope:target.province_definition = {
			any_location_in_province_definition = {
				has_owner = no
				is_ownable = yes
				is_land = yes
				any_neighbor_location = {
					owner ?= root
				}
			}
		}

	}

	illustration_tags = {
		10 = happy
		10 = exterior
	}

	immediate = {
		scope:target.province_definition = {
			random_location_in_province_definition = {
				limit = { 
					has_owner = no
					is_ownable = yes
					is_land = yes
					any_neighbor_location = {
						owner ?= root
					}
				}
				save_scope_as = new_colony
			}
		}

	}
	
	
	
	option = {
		name = colonial_charter.1.a
		scope:new_colony = {
			add_pop = {
				culture = root.culture
				religion = root.religion
				type = pop_type:peasants
				size = 0.2
			}
		}
	}

	option = {
		name = colonial_charter.1.b
		scope:new_colony = {
			add_pop = {
				culture = root.culture
				religion = root.religion
				type = pop_type:peasants
				size = 0.05
			}
		}
		add_gold = {
			value = {
				scope:target.province_definition = {
					every_location_in_province_definition = {
						add = this.population
					}
				}
				add = "capital.distance_to(scope:new_colony)"
			}
			divide = 100
			multiply = 10
		}
	}
}

#scope:target is the Province of the charter
colonial_charter.100 = {
	hide_portraits = yes
	type = country_event	

	title = colonial_charter.100.title
	desc = colonial_charter.100.desc
	outcome = neutral

	immediate = {
		save_scope_as = actor
		#Select the best location for a capital
		if = {
			limit = {
				scope:target = {
					any_location_in_province = {
						is_city = yes
						count = 0
					}
				}
			}
			scope:target = {
				ordered_location_in_province = {
					max = 1
					order_by = best_capital_for_colony
					
					save_scope_as = new_town
				}
			}
			
		}
		#If already there is a colony on our area, save it
		ordered_subject = {
			limit = {
				is_subject_type = colonial_nation
				capital = {
					area = scope:target.area
				}
			}

			order_by = total_population
			max = 1
			save_scope_as = scope_area_subject
		}
		#If already there is a colony on our area or an adjacent one, save it
		ordered_subject = {
			limit = {
				is_subject_type = colonial_nation
				capital = {
					OR = {
						area = scope:target.area
						area = {
							any_neighbor_area = {
								this = scope:target.area
							}
						}
					}
				}
				NAND = { exists = scope:scope_area_subject	this = scope:scope_area_subject }
			}

			order_by = total_population
			max = 1
			save_scope_as = scope_adjacent_area_subject
		}


		#Get a colonial nation vassal neighboring the target province
		if = {
			limit = {
				scope:target = {
					any_location_in_province = {
						any_coast_border_location = {
							has_owner = yes
							owner = {
								is_subject_of = ROOT
								is_subject_type = colonial_nation
								NAND = { exists = scope:scope_area_subject	this = scope:scope_area_subject }
								NAND = { exists = scope:scope_adjacent_area_subject	this = scope:scope_adjacent_area_subject }
							}
						}
					}
				}
			}
			scope:target = {
				random_location_in_province = {
					limit = {
						any_coast_border_location = {
							has_owner = yes
							owner = {
								is_subject_of = ROOT
								is_subject_type = colonial_nation
								NAND = { exists = scope:scope_area_subject	this = scope:scope_area_subject }
								NAND = { exists = scope:scope_adjacent_area_subject	this = scope:scope_adjacent_area_subject }
							}
						}
					}
					random_coast_border_location = {
						limit = {
							has_owner = yes
							owner = {
								is_subject_of = ROOT
								is_subject_type = colonial_nation
								NAND = { exists = scope:scope_area_subject	this = scope:scope_area_subject }
								NAND = { exists = scope:scope_adjacent_area_subject	this = scope:scope_adjacent_area_subject }
							}
						}
						owner = { save_scope_as = first_neighboring_country }
					}
				}
			}
		}

		#Get a second colonial nation vassal neighboring the target province, different from first_neighboring_country
		if = {
			limit = {
				scope:target = {
					any_location_in_province = {
						any_coast_border_location = {
							has_owner = yes
							owner = {
								is_subject_of = ROOT
								is_subject_type = colonial_nation
								NAND = { exists = scope:scope_area_subject	this = scope:scope_area_subject }
								NAND = { exists = scope:scope_adjacent_area_subject	this = scope:scope_adjacent_area_subject }
								NAND = { exists = scope:first_neighboring_country	this = scope:first_neighboring_country }
							}
						}
					}
				}
			}
			scope:target = {
				random_location_in_province = {
					limit = {
						any_coast_border_location = {
							has_owner = yes
							owner = {
								is_subject_of = ROOT
								is_subject_type = colonial_nation
								NAND = { exists = scope:scope_area_subject	this = scope:scope_area_subject }
								NAND = { exists = scope:scope_adjacent_area_subject	this = scope:scope_adjacent_area_subject }
								NAND = { exists = scope:first_neighboring_country	this = scope:first_neighboring_country }
							}
						}
					}
					random_coast_border_location = {
						limit = {
							has_owner = yes
							owner = {
								is_subject_of = ROOT
								is_subject_type = colonial_nation
								NAND = { exists = scope:scope_area_subject	this = scope:scope_area_subject }
								NAND = { exists = scope:scope_adjacent_area_subject	this = scope:scope_adjacent_area_subject }
								NAND = { exists = scope:first_neighboring_country	this = scope:first_neighboring_country }
							}
						}
						owner = { save_scope_as = second_neighboring_country }
					}
				}
			}
		}

		#Get the most populous colonial nation vassal with capital in the same region, different from first and second neighboring
		ordered_subject = {
			limit = {
				is_subject_type = colonial_nation
				capital = { region = scope:target.region }
				NAND = { exists = scope:scope_area_subject	this = scope:scope_area_subject }
				NAND = { exists = scope:scope_adjacent_area_subject	this = scope:scope_adjacent_area_subject }
				NAND = { exists = scope:first_neighboring_country	this = scope:first_neighboring_country }
				NAND = { exists = scope:second_neighboring_country	this = scope:second_neighboring_country }
			}
			order_by = total_population
			max = 1
			save_scope_as = regional_subject
		}

		#Save the Province Capital for the event image
		scope:target = {
			capital = { save_scope_as = target_location }
		}
	}
	
	option = { #Form a new Colonial Subject
		name = colonial_charter.100.a
		trigger = {
			scope:target = { is_overseas_for_owner = yes  }
		}
		
		if = {
			limit = {
				exists = scope:new_town
			}
			scope:new_town = {
				change_location_rank_effect = { location_rank = location_rank:town }
			}
		}
		
		scope:target = { 
			create_location_country_from_province = {
				subject_type = subject_type:colonial_nation
			
				hidden_effect = {
					setup_colonial_nation = yes
				}
			}
		}
		
		ai_will_select = {
			value = 0
			if = { #13 colonies exception
				limit = {
					scope:target = {
						is_in_13_colonies = yes
					}
				}
				if = {#If there is already a colony in the area, never create a new one
					limit = {
						exists = scope:scope_area_subject
					}
					add = -1000
				}
				else = {
					#If we are part of the 13 colonies and no other colony exist in our area, we stop caring about anything, we make our area colony
					add = 10000
				}
			
			}
			if = { #If any neighboring or regional colonial nation exists, never create a new one
				limit = {
					exists = scope:scope_adjacent_area_subject
				}
				add = -1000
			}
			else = { #No existing colonial nation to join, forming a new one is preferred
				add = 1000
			}
		}
	}
	
	option = { #Join it to scope_area_subject
		name = colonial_charter.100.h
		trigger = {
			exists = scope:scope_area_subject
		}

		if = {
			limit = {
				exists = scope:new_town
			}
			scope:new_town = {
				change_location_rank_effect = { location_rank = location_rank:town }
				cc_setup_new_town = yes
			}
		}

		scope:target = {
			change_province_owner = scope:scope_area_subject
		}

		ai_will_select = {
			value = 3000
		}
	}
	
	option = { #Join it to scope_adjacent_area_subject
		name = colonial_charter.100.g
		trigger = {
			exists = scope:scope_adjacent_area_subject
		}

		if = {
			limit = {
				exists = scope:new_town
			}
			scope:new_town = {
				change_location_rank_effect = { location_rank = location_rank:town }
				cc_setup_new_town = yes
			}
		}

		scope:target = {
			change_province_owner = scope:scope_adjacent_area_subject 
		}

		ai_will_select = {
			value = 2000
			if = { #13 colonies exception
				limit = {
					scope:target = {
						is_in_13_colonies = yes
					}
				}
				if = {#If there is already a colony in the area, join it
					limit = {
						scope:scope_adjacent_area_subject = {
							capital.area = scope:target.area
						}
					}
					value = 2000
				}
				else = {
					#If we are part of the 13 colonies and scope_adjacent_area_subject does not have its capital in our area we don't join
					value = -2000
				}
			
			}
			else_if = {#Secondary 13 colonies exception
				limit = { # If scope_adjacent_area_subject is a 13 colonies colony we won't join to it either
					scope:scope_adjacent_area_subject.capital = {
						is_in_13_colonies = yes
					}
				}
				value = -2000
			}
		}
	}

	option = { #Join it to first_neighboring_country
		name = colonial_charter.100.b
		trigger = {
			exists = scope:first_neighboring_country
		}

		if = {
			limit = {
				exists = scope:new_town
			}
			scope:new_town = {
				change_location_rank_effect = { location_rank = location_rank:town }
				cc_setup_new_town = yes
			}
		}

		scope:target = {
			change_province_owner = scope:first_neighboring_country
		}

		ai_will_select = {
			if = { #13 colonies exception
				limit = {
					scope:target = {
						is_in_13_colonies = yes
					}
				}
				if = {#If first_neighboring_country is already a colony in the area, join it
					limit = {						
						scope:first_neighboring_country = {
							capital.area = scope:target.area
						}
					}
					value = 3000
				}
				else = {
					#If we are part of the 13 colonies and first_neighboring_country does not have its capital in our area we don't join
					value = -1000
				}
			
			}
			else_if = {#Secondary 13 colonies exception
				limit = { # If first_neighboring_country is a 13 colonies colony we won't join to it either
					scope:first_neighboring_country.capital = {
						is_in_13_colonies = yes
					}
				}
				value = -1000
			}
			#If the first_neighboring_country has its capital in the same Area (or neighboring ones) we prioritize joining to this one
			else_if = {
				limit = {
					scope:first_neighboring_country = {
						capital = {
							area = {
								OR = {
									this = scope:target.area
									any_neighbor_area = {
										this = scope:target.area
									}
								}
							}
						}
					}
				}
				value = 3000
			}
			else = {
				value = -1000
			}
		}
	}

	option = { #Join it to second_neighboring_country
		name = colonial_charter.100.c
		trigger = {
			exists = scope:second_neighboring_country
		}

		if = {
			limit = {
				exists = scope:new_town
			}
			scope:new_town = {
				change_location_rank_effect = { location_rank = location_rank:town }
				cc_setup_new_town = yes
			}
		}

		scope:target = {
			change_province_owner = scope:second_neighboring_country
		}

		ai_will_select = {
			if = { #13 colonies exception
				limit = {
					scope:target = {
						is_in_13_colonies = yes
					}
				}
				if = {#If second_neighboring_country is already a colony in the area, join it
					limit = {						
						scope:second_neighboring_country = {
							capital.area = scope:target.area
						}
					}
					value = 2500
				}
				else = {
					#If we are part of the 13 colonies and second_neighboring_country does not have its capital in our area we don't join
					value = -500
				}
			
			}
			else_if = {#Secondary 13 colonies exception
				limit = { # If second_neighboring_country is a 13 colonies colony we won't join to it either
					scope:second_neighboring_country.capital = {
						is_in_13_colonies = yes
					}
				}
				value = -500
			}
			#If the second_neighboring_country has its capital in the same Area (or neighboring ones) we prioritize joining to this one
			else_if = {
				limit = {
					scope:second_neighboring_country = {
						capital = {
							OR = {
								area = scope:target.area
								area = {
									any_neighbor_area = {
										this = scope:target.area
									}
								}
							}
						}
					}
				}
				value = 2500
			}
			else = {
				value = -500
			}
		}
	}

	option = { #Join it to regional_subject
		name = colonial_charter.100.f

		trigger = {
			exists = scope:regional_subject
		}

		if = {
			limit = {
				exists = scope:new_town
			}
			scope:new_town = {
				change_location_rank_effect = { location_rank = location_rank:town }
				cc_setup_new_town = yes
			}
		}

		scope:target = {
			change_province_owner = scope:regional_subject
		}

		ai_will_select = {
			value = -250
		}
	}

	option = { #Keep it as part of the Metropolis
		name = colonial_charter.100.d
		
		scope:target = { 
			every_location_in_province = {
				limit = {
					integration_level = colonized
				}
				change_integration_level = integrated
			}
		}
		

		ai_chance = {
			factor = 1
			modifier = {
				trigger = {
					scope:target = { is_overseas_for_owner = yes }
				}
				factor = -1
			}
		}
	}

	option = { #Play as the Colony
		name = colonial_charter.100.e
		high_risk_option = yes
		trigger = {
			is_human = yes
			scope:target = { is_overseas_for_owner = yes  }
		}
		if = {
			limit = {
				exists = scope:new_town
			}
			scope:new_town = {
				change_location_rank_effect = { location_rank = location_rank:town }
			}
		}
		scope:target = { 
			create_location_country_from_province = {
				subject_type = subject_type:colonial_nation
				hidden_effect = {
					setup_colonial_nation = yes
				}
				save_scope_as = target_country
			}
		}
		custom_tooltip = {
			text = colonial_charter.100.e.tt
			change_player = scope:target_country
		}

		# needed when ai and human is playing together
		ai_chance = {
			factor = -1
		}
	}

	after = {
		trigger_event_silently = treaty_of_tordesillas.1
	}
	trigger = {
		current_date > 476.9.4
	}
}
