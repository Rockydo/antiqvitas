royal_monopoly_on_trade = {
	estate = crown_estate

	modifier_when_in_debate = {
		trade_income = 0.05
	}

	allow = {
		"estate_power(estate_type:burghers_estate)" < 0.5
		at_war = no
	}

	chance = {
		add = 1
		add = {
			if = {
				limit = { "estate_power(estate_type:burghers_estate)" < 0.3 }
				value = 2
			}
		}
	}

	on_debate_passed = {
		add_country_modifier = {
			modifier = parl_royal_monopoly_mod
			years = 10
			mode = add_and_extend
		}
		change_societal_value = { type = mercantilism_vs_free_trade value = societal_value_tiny_move_to_left }
	}

	on_debate_failed = {
		add_estate_satisfaction = { type = estate_type:burghers_estate value = estate_satisfaction_extreme_bonus }
		parliament_debate_failed_effect = yes
	}
}
